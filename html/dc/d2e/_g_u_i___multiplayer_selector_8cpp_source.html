78 RoR::GUI::MpServerlistData* FetchServerlist(std::string portal_url)
79 {
80  std::string serverlist_url = portal_url + "/server-list?json=true";
81  std::string response_payload;
82  std::string response_header;
83  long response_code = 0;
84 
85  CURL *curl = curl_easy_init();
86  curl_easy_setopt(curl, CURLOPT_URL, serverlist_url.c_str());
87  curl_easy_setopt(curl, CURLOPT_NOPROGRESS, 1L);
88  curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, CurlWriteFunc);
89  curl_easy_setopt(curl, CURLOPT_WRITEDATA, &response_payload);
90  curl_easy_setopt(curl, CURLOPT_HEADERDATA, &response_header);
91 
92  curl_easy_perform(curl);
93  curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &response_code);
94 
95  curl_easy_cleanup(curl);
96  curl = nullptr;
97 
98  RoR::GUI::MpServerlistData* res = new RoR::GUI::MpServerlistData();
99  if (response_code != 200)
100  {
101  Ogre::LogManager::getSingleton().stream()
102  << "[RoR|Multiplayer] Failed to retrieve serverlist; HTTP status code: " << response_code;
103  res->message = "Error connecting to server :(";
104  return res;
105  }
106 
107  rapidjson::Document j_data_doc;
108  j_data_doc.Parse(response_payload.c_str());
109  if (j_data_doc.HasParseError() || !j_data_doc.IsArray())
110  {
111  Ogre::LogManager::getSingleton().stream()
112  << "[RoR|Multiplayer] Error parsing serverlist JSON"; // TODO: Report the actual error
113  res->message = "Server returned invalid data :(";
114  return res;
115  }
116 
117  // Pre-process data for display
118  size_t num_rows = j_data_doc.GetArray().Size();
119  res->servers.resize(num_rows);
120  for (size_t i = 0; i < num_rows; ++i)
121  {
122  rapidjson::Value& j_row = j_data_doc[i];
123 
124  res->servers[i].display_name = j_row["name"].GetString();
125  res->servers[i].display_terrn = j_row["terrain-name"].GetString();
126  res->servers[i].net_host = j_row["ip"].GetString();
127  res->servers[i].net_port = j_row["port"].GetInt();
128 
129  const bool has_pw = j_row["has-password"].GetBool();
130  res->servers[i].has_password = has_pw;
131  res->servers[i].display_passwd = (has_pw) ? "Yes" : "No";
132 
133  char display_host[400];
134  snprintf(display_host, 400, "%s:%d", j_row["ip"].GetString(), j_row["port"].GetInt());
135  res->servers[i].display_host = display_host;
136 
137  char display_users[200];
138  snprintf(display_users, 200, "%d / %d", j_row["current-users"].GetInt(), j_row["max-clients"].GetInt());
139  res->servers[i].display_users = display_users;
140  }
141 
142  res->success = true;
143  return res;
144 }
145 #endif // defined(USE_CURL)

396  m_is_refreshing = false;
