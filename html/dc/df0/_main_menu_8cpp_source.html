88 void MainMenu::EnterMainMenuLoop()
89 {
90  RoRWindowEventUtilities::addWindowEventListener(App::GetOgreSubsystem()->GetRenderWindow(), this);
91 
92  // ==== FPS-limiter ====
93  // TODO: Is this necessary in menu?
94 
95  unsigned long timeSinceLastFrame = 1;
96  unsigned long startTime = 0;
97  unsigned long minTimePerFrame = 0;
98  unsigned long fpsLimit = App::gfx_fps_limit.GetActive(); // TOD: use GVar directly without copying
99 
100  if (fpsLimit < 10 || fpsLimit >= 200)
101  {
102  fpsLimit = 0;
103  }
104 
105  if (fpsLimit)
106  {
107  minTimePerFrame = 1000 / fpsLimit;
108  }
109 
110  App::GetOgreSubsystem()->GetOgreRoot()->addFrameListener(this);
111 
112  while (App::app_state.GetPending() == AppState::MAIN_MENU)
113  {
114  startTime = App::GetOgreSubsystem()->GetTimer()->getMilliseconds();
115 
116  this->MainMenuLoopUpdate(static_cast<float>(timeSinceLastFrame)/1000);
117 
118  if (RoR::App::GetGuiManager()->GetMainSelector()->IsFinishedSelecting())
119  {
120  CacheEntry* selected_map = RoR::App::GetGuiManager()->GetMainSelector()->GetSelectedEntry();
121  if (selected_map != nullptr)
122  {
123  App::GetGuiManager()->GetMainSelector()->Reset(); // TODO: Eliminate this mechanism ~ only_a_ptr 09/2016
124  App::app_state.SetPending(AppState::SIMULATION);
125  App::sim_terrain_name.SetPending(selected_map->fname.c_str());
126  }
127  }
128 
129 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32 || OGRE_PLATFORM == OGRE_PLATFORM_LINUX
130  RoRWindowEventUtilities::messagePump();
131 #endif
132  Ogre::RenderWindow* rw = RoR::App::GetOgreSubsystem()->GetRenderWindow();
133  if (rw->isClosed())
134  {
135  App::app_state.SetPending(AppState::SHUTDOWN);
136  continue;
137  }
138 
139  App::GetGuiManager()->NewImGuiFrame(static_cast<float>(timeSinceLastFrame)/1000);
140  App::GetGuiManager()->DrawMainMenuGui();
141 
142  App::GetOgreSubsystem()->GetOgreRoot()->renderOneFrame();
143 
144 #ifdef USE_SOCKETW
145  if ((App::mp_state.GetActive() == MpState::CONNECTED) && RoR::Networking::CheckError())
146  {
147  Ogre::String title = Ogre::UTFString(_L("Network fatal error: ")).asUTF8();
148  Ogre::String msg = RoR::Networking::GetErrorMessage().asUTF8();
149  App::GetGuiManager()->ShowMessageBox(title, msg, true, "OK", true, false, "");
150  App::app_state.SetPending(AppState::MAIN_MENU);
151 
152  RoR::App::GetGuiManager()->GetMainSelector()->Hide();
153  RoR::App::GetGuiManager()->GetMainSelector()->Show(LT_Terrain);
154  }
155 #endif
156 
157  if (!rw->isActive() && rw->isVisible())
158  rw->update(); // update even when in background !
159 
160  // FPS-limiter. TODO: Is this necessary in menu?
161  if (fpsLimit && timeSinceLastFrame < minTimePerFrame)
162  {
163  // Sleep twice as long as we were too fast.
164  int ms = static_cast<int>((minTimePerFrame - timeSinceLastFrame) << 1);
165  std::this_thread::sleep_for(std::chrono::milliseconds(ms));
166  }
167 
168  timeSinceLastFrame = RoR::App::GetOgreSubsystem()->GetTimer()->getMilliseconds() - startTime;
169  }
170  RoRWindowEventUtilities::removeWindowEventListener(App::GetOgreSubsystem()->GetRenderWindow(), this);
171  App::GetOgreSubsystem()->GetOgreRoot()->removeFrameListener(this);
172 }

174 void MainMenu::MainMenuLoopUpdate(float seconds_since_last_frame)
175 {
176  if (seconds_since_last_frame == 0)
177  {
178  return;
179  }
180  if (seconds_since_last_frame > 1.0 / 20.0) // TODO: Does this have any sense for menu?
181  {
182  seconds_since_last_frame = 1.0 / 20.0;
183  }
184 
185  if (RoR::App::GetOgreSubsystem()->GetRenderWindow()->isClosed())
186  {
187  App::app_state.SetPending(AppState::SHUTDOWN);
188  return;
189  }
190 
191  auto gui = App::GetGuiManager();
192 
193 #ifdef USE_SOCKETW
194  if (App::mp_state.GetActive() == MpState::CONNECTED)
195  {
196  gui->GetMpClientList()->update();
197  }
198  else if (App::mp_state.GetPending() == MpState::CONNECTED)
199  {
200  Networking::ConnectState con_state = Networking::CheckConnectingState();
201  if (con_state == Networking::ConnectState::IDLE) // Not connecting yet
202  {
203  gui->SetVisible_MultiplayerSelector(false);
204  bool connect_started = Networking::StartConnecting();
205  gui->SetVisible_GameMainMenu(!connect_started);
206  if (!connect_started)
207  {
208  App::GetGuiManager()->ShowMessageBox("Multiplayer: connection failed", Networking::GetErrorMessage().asUTF8_c_str(), true, "OK", false, false, "");
209  App::mp_state.SetActive(RoR::MpState::DISABLED);
210  }
211  }
212  else if (con_state == Networking::ConnectState::FAILURE) // Just failed (only returned once)
213  {
214  App::mp_state.SetActive(RoR::MpState::DISABLED);
215  App::GetGuiManager()->ShowMessageBox("Multiplayer: connection failed", Networking::GetErrorMessage().asUTF8_c_str(), true, "OK", false, false, "");
216  App::GetGuiManager()->SetVisible_GameMainMenu(true);
217  }
218  else if (con_state == Networking::ConnectState::SUCCESS) // Just succeeded (only returned once)
219  {
220  App::mp_state.SetActive(RoR::MpState::CONNECTED);
221  gui->SetVisible_MpClientList(true);
222  ChatSystem::SendStreamSetup();
223  App::CheckAndCreateMumble();
224  String terrain_name = Networking::GetTerrainName();
225  if (terrain_name != "any")
226  {
227  App::sim_terrain_name.SetPending(terrain_name.c_str());
228  App::app_state.SetPending(AppState::SIMULATION);
229  }
230  else
231  {
232  // Connected -> go directly to map selector
233  if (App::diag_preset_terrain.IsActiveEmpty())
234  {
235  gui->GetMainSelector()->Reset();
236  gui->GetMainSelector()->Show(LT_Terrain);
237  }
238  else
239  {
240  App::app_state.SetPending(AppState::SIMULATION);
241  }
242  }
243  }
244  }
245 
246  if (App::GetGuiManager()->GetMpSelector()->IsRefreshThreadRunning())
247  {
248  App::GetGuiManager()->GetMpSelector()->CheckAndProcessRefreshResult();
249  }
250 #endif // USE_SOCKETW
251 
252  RoR::App::GetInputEngine()->Capture();
253 
254  MainMenuLoopUpdateEvents(seconds_since_last_frame);
255 
256 #ifdef USE_ANGELSCRIPT
257  ScriptEngine::getSingleton().framestep(seconds_since_last_frame);
258 #endif
259 }

261 void MainMenu::MainMenuLoopUpdateEvents(float seconds_since_last_frame)
262 {
263  if (seconds_since_last_frame == 0.0f)
264  {
265  return;
266  }
267 
268  RoR::App::GetInputEngine()->updateKeyBounces(seconds_since_last_frame);
269 
270  if (! RoR::App::GetInputEngine()->getInputsChanged())
271  {
272  return;
273  }
274 
275  if (RoR::App::GetOverlayWrapper() != nullptr)
276  {
277  RoR::App::GetOverlayWrapper()->update(seconds_since_last_frame); // TODO: What's the meaning of this? It only updates some internal timer. ~ only_a_ptr
278  }
279 
280  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_COMMON_QUIT_GAME)) // TODO: EV_COMMON_QUIT_GAME should really be EV_COMMON_ESCAPE ~ only_a_ptr, 12/2017
281  {
282  if (App::GetGuiManager()->IsVisible_GameAbout())
283  {
284  App::GetGuiManager()->SetVisible_GameAbout(false);
285  App::GetGuiManager()->SetVisible_GameMainMenu(true);
286  }
287  else if (App::GetGuiManager()->IsVisible_MainSelector())
288  {
289  App::GetGuiManager()->GetMainSelector()->Cancel();
290  App::GetGuiManager()->SetVisible_GameMainMenu(true);
291  }
292  else if (App::GetGuiManager()->IsVisible_GameSettings())
293  {
294  App::GetGuiManager()->SetVisible_GameSettings(false);
295  App::GetGuiManager()->SetVisible_GameMainMenu(true);
296  }
297  else if (App::GetGuiManager()->IsVisible_MultiplayerSelector())
298  {
299  App::GetGuiManager()->SetVisible_MultiplayerSelector(false);
300  App::GetGuiManager()->SetVisible_GameMainMenu(true);
301  }
302  else
303  {
304  App::app_state.SetPending(AppState::SHUTDOWN);
305  }
306  return;
307  }
308 
309  auto gui_man = App::GetGuiManager();
310  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_COMMON_CONSOLE_TOGGLE, 5.f))
311  {
312  gui_man->SetVisible_Console(!gui_man->IsVisible_Console());
313  }
314 
315  // TODO: screenshot
316 
317  // FOV settings disabled in menu
318 
319  // FIXME: full screen/windowed screen switching
320  //if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_COMMON_FULLSCREEN_TOGGLE, 2.0f)) {}
321 }
