63 void SkyManager::DetectSkyUpdate()
64 {
65  if (!m_caelum_system || !gEnv->terrainManager)
66  return;
67  Caelum::LongReal c = m_caelum_system->getUniversalClock()->getJulianDay();
68 
69  if (c - m_last_clock > 0.001f)
70  {
71  TerrainGeometryManager* gm = gEnv->terrainManager->getGeometryManager();
72  if (gm)
73  gm->updateLightMap();
74  }
75 
76  m_last_clock = c;
77 }

79 void SkyManager::LoadCaelumScript(std::string script, int fogStart, int fogEnd)
80 {
81  // load the caelum config
82  try
83  {
84  Caelum::CaelumPlugin::getSingleton().loadCaelumSystemFromScript(m_caelum_system, script);
85 
86  // overwrite some settings
87 #ifdef CAELUM_VERSION_SEC
88  // important: overwrite fog settings if not using infinite farclip
89  if (fogStart != -1 && fogEnd != -1 && fogStart < fogEnd)
90  {
91  // setting farclip (hacky)
92  gEnv->mainCamera->setFarClipDistance(fogEnd / 0.8);
93  // custom boundaries
94  m_caelum_system->setManageSceneFog(Ogre::FOG_LINEAR);
95  m_caelum_system->setManageSceneFogStart(fogStart);
96  m_caelum_system->setManageSceneFogEnd(fogEnd);
97  }
98  else if (gEnv->mainCamera->getFarClipDistance() > 0)
99  {
100  if (fogStart != -1 && fogEnd != -1)
101  {
102  LOG("CaelumFogStart must be smaller then CaelumFogEnd. Ignoring boundaries.");
103  }
104  else if (fogStart != -1 || fogEnd != -1)
105  {
106  LOG("You always need to define both boundaries (CaelumFogStart AND CaelumFogEnd). Ignoring boundaries.");
107  }
108  // non infinite farclip
109  float farclip = gEnv->mainCamera->getFarClipDistance();
110  m_caelum_system->setManageSceneFog(Ogre::FOG_LINEAR);
111  m_caelum_system->setManageSceneFogStart(farclip * 0.7);
112  m_caelum_system->setManageSceneFogEnd(farclip * 0.9);
113  }
114  else
115  {
116  // no fog in infinite farclip
117  m_caelum_system->setManageSceneFog(Ogre::FOG_NONE);
118  }
119 #else
120 #error please use a recent Caelum version, see http://www.rigsofrods.org/wiki/pages/Compiling_3rd_party_libraries#Caelum
121 #endif // CAELUM_VERSION
122  // now optimize the moon a bit
123  if (m_caelum_system->getMoon())
124  {
125  m_caelum_system->getMoon()->setAutoDisable(true);
126  //m_caelum_system->getMoon()->setAutoDisableThreshold(1);
127  m_caelum_system->getMoon()->setForceDisable(true);
128  m_caelum_system->getMoon()->getMainLight()->setCastShadows(false);
129  }
130 
131  m_caelum_system->setEnsureSingleShadowSource(true);
132  m_caelum_system->setEnsureSingleLightSource(true);
133 
134  // enforcing update, so shadows are set correctly before creating the terrain
135  m_caelum_system->updateSubcomponents(0.1);
136  }
137  catch (Ogre::Exception& e)
138  {
139  RoR::LogFormat("[RoR] Exception while loading sky script: %s", e.getFullDescription().c_str());
140  }
141  Ogre::Vector3 lightsrc = m_caelum_system->getSun()->getMainLight()->getDirection();
142  m_caelum_system->getSun()->getMainLight()->setDirection(lightsrc.normalisedCopy());
143 }
