100 bool RoR::OTCParser::LoadPageConfig(Ogre::DataStreamPtr &ds, RoR::OTCPage& page, const char* filename)
101 {
102  const int LINE_BUF_LEN = 4000;
103  char line_buf[LINE_BUF_LEN];
104  try
105  {
106  ds->readLine(line_buf, LINE_BUF_LEN);
107  page.heightmap_filename = RoR::Utils::SanitizeUtf8CString(line_buf);
108  RoR::Utils::TrimStr(page.heightmap_filename);
109 
110  ds->readLine(line_buf, LINE_BUF_LEN);
111  page.num_layers = PARSEINT(line_buf);
112 
113  while (!ds->eof())
114  {
115  size_t len = ds->readLine(line_buf, LINE_BUF_LEN);
116  if (len == 0 || line_buf[0] == ';' || line_buf[0] == '/') { continue; }
117 
118  std::string line_sane = RoR::Utils::SanitizeUtf8String(std::string(line_buf));
119  Ogre::StringVector args = Ogre::StringUtil::split(line_sane, ",");
120  if (args.size() < 3)
121  {
122  LOG(std::string("[RoR|Terrain] Invalid OTC page config: [") + filename + "]");
123  return false;
124  }
125 
126  OTCLayer layer;
127  layer.world_size = PARSEREAL(args[0]);
128  layer.diffusespecular_filename = Utils::TrimStr(args[1]);
129  layer.normalheight_filename = Utils::TrimStr(args[2]);
130  if (args.size() > 3)
131  {
132  layer.blendmap_filename = Utils::TrimStr(args[3]);
133  }
134  if (args.size() > 4)
135  {
136  layer.blend_mode = Utils::TrimStr(args[4])[0];
137  }
138  if (args.size() > 5)
139  {
140  layer.alpha = PARSEREAL(args[5]);
141  }
142 
143  page.layers.push_back(layer);
144  }
145 
146  if (page.heightmap_filename.find(".raw") != std::string::npos)
147  {
148  page.is_heightmap_raw = true;
149  }
150 
151  int actual_num_layers = static_cast<int>(page.layers.size());
152  if (page.num_layers > actual_num_layers)
153  {
154  LogFormat("[RoR|Terrain] Warning: File \"%s\" declares %d layers but defines only %d. Correcting declared layer count to %d",
155  filename, page.num_layers, actual_num_layers, actual_num_layers);
156  page.num_layers = actual_num_layers;
157  }
158  }
159  catch (...)
160  {
161  this->HandleException(filename);
162  return false;
163  }
164  return true;
165 }
