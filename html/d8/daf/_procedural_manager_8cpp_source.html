59 int ProceduralManager::updateObject(ProceduralObject& po)
60 {
61  if (po.loadingState == 1 && po.road)
62  deleteObject(po);
63  // create new road2 object
64  po.road = new Road2(objectcounter++);;
65 
66  std::vector<ProceduralPoint>::iterator it;
67  for (it = po.points.begin(); it != po.points.end(); it++)
68  {
69  ProceduralPoint pp = *it;
70  po.road->addBlock(pp.position, pp.rotation, pp.type, pp.width, pp.bwidth, pp.bheight, pp.pillartype);
71  }
72  po.road->finish();
73 
74  po.loadingState = 1;
75  return 0;
76 }
