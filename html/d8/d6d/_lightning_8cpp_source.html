88  Ogre::Real delta;
89  Ogre::Vector2 bounds;
90  Ogre::Billboard* bb;
91  for(Ogre::uint32 k = 0; k < mSegments.size(); k++)
92  {

181  void Lightning::update(Ogre::Real timeSinceLastFrame)
182  {
183  if (!mCreated)
184  {
185  return;
186  }
187 
188  timeSinceLastFrame *= mTimeMultiplier;
189 
190  // mTime timeline: (Note: Time multipliers are random)
191  // 0.0 -> 1.2 : Ray creation(fordward effect) + Big flash
192  // 1.2 -> 2.0 : Sinus flashing pattern
193  // 2.0 -> 3.0 Ray fading
194 
195  Ogre::Real alpha = 0.5f;
196  Ogre::Real maxAlpha = 1.5f;
197 
198  if (mTime < 1)
199  {
200  mTime += timeSinceLastFrame*mTimeMultipliers.x;
201 
202  if (mTime > 2) mTime = 1.5f; // Prevent big changes
203 
204  if (mTime > 0.8f) // Big flash start
205  {
206  alpha += (mTime-0.8f)*(maxAlpha/0.2f);
207  }
208  }
209  else if (mTime > 1 && mTime < 2)
210  {
211  mTime += timeSinceLastFrame*mTimeMultipliers.y;
212 
213  if (mTime > 3) mTime = 2.5f; // Prevent big changes
214 
215  if (mTime < 1.2f) // Big flash end
216  {
217  alpha += (0.2f-(mTime-1.0f))*(maxAlpha/0.2f);
218  }
219  else // Sinus flashing pattern
220  {
221  alpha += Ogre::Math::Abs(Ogre::Math::Sin((mTime-1.2f)*1.5f*mTimeMultipliers.x));
222  }
223  }
224  else if (mTime > 2) // Ray fading
225  {
226  mTime += timeSinceLastFrame*mTimeMultipliers.z;
227 
228  if (mTime > 3)
229  {
230  mTime = 3; // Prevent big changes
231  mFinished = true;
232  }
233 
234  alpha += Ogre::Math::Abs(Ogre::Math::Sin((2-1.2f)*1.5f*mTimeMultipliers.x));
235  alpha *= 3.0f-mTime;
236  }
237 
238  mIntensity = alpha;
239 
240  _updateData(alpha, mTime > 1 ? 1 : mTime, mTime);
241  }

254  {
