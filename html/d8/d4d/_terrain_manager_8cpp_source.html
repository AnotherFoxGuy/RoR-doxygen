140 void TerrainManager::loadTerrain(String filename)
141 {
142  DataStreamPtr ds;
143 
144  try
145  {
146  String group = ResourceGroupManager::getSingleton().findGroupContainingResource(filename);
147  ds = ResourceGroupManager::getSingleton().openResource(filename, group);
148  }
149  catch (...)
150  {
151  LOG("[RoR|Terrain] File not found: " + filename);
152  return;
153  }
154 
155  PROGRESS_WINDOW(10, _L("Loading Terrain Configuration"));
156 
157  LOG(" ===== LOADING TERRAIN " + filename);
158 
159  Terrn2Parser parser;
160  if (! parser.LoadTerrn2(m_def, ds))
161  {
162  LOG("[RoR|Terrain] Failed to parse: " + filename);
163  for (std::string msg : parser.GetMessages())
164  {
165  LOG("[RoR|Terrain] \tMessage: " + msg);
166  }
167  return;
168  }
169  gravity = m_def.gravity;
170 
171  // then, init the subsystems, order is important :)
172  initSubSystems();
173 
174  fixCompositorClearColor();
175 
176  LOG(" ===== LOADING TERRAIN GEOMETRY " + filename);
177 
178  // load the terrain geometry
179  PROGRESS_WINDOW(80, _L("Loading Terrain Geometry"));
180  geometry_manager->InitTerrain(m_def.ogre_ter_conf_filename);
181 
182  LOG(" ===== LOADING TERRAIN WATER " + filename);
183  // must happen here
184  initWater();
185 
186  LOG(" ===== LOADING TERRAIN OBJECTS " + filename);
187 
188  PROGRESS_WINDOW(90, _L("Loading Terrain Objects"));
189  loadTerrainObjects();
190 
191  // bake the decals
192  //finishTerrainDecal();
193 
194  // init things after loading the terrain
195  initTerrainCollisions();
196 
197  // init the survey map
198  if (!RoR::App::gfx_minimap_disabled.GetActive())
199  {
200  PROGRESS_WINDOW(45, _L("Initializing Overview Map Subsystem"));
201  m_survey_map = new SurveyMapManager();
202  }
203 
204  PROGRESS_WINDOW(95, _L("Initializing terrain light properties"));
205  geometry_manager->UpdateMainLightPosition(); // Initial update takes a while
206  collisions->finishLoadingTerrain();
207  LOG(" ===== TERRAIN LOADING DONE " + filename);
208 
209  // causing visual ground glitches
210 /* if (App::gfx_sky_mode.GetActive() == GfxSkyMode::SKYX)
211  {
212  TerrainGroup::TerrainIterator ti = geometry_manager->getTerrainGroup()->getTerrainIterator();
213  while (ti.hasMoreElements())
214  {
215  Terrain* t = ti.getNext()->instance;
216  MaterialPtr ptr = t->getMaterial();
217  gEnv->SkyX->GetSkyX()->getGPUManager()->addGroundPass(
218  static_cast<Ogre::MaterialPtr>(ptr)->getTechnique(0)->createPass(), 5000, Ogre::SBT_TRANSPARENT_COLOUR);
219  }
220  }*/

224 void TerrainManager::initSubSystems()
225 {

232  // objects - .odef support
233  PROGRESS_WINDOW(17, _L("Initializing Object Subsystem"));
234  initObjects();
235 
236  PROGRESS_WINDOW(19, _L("Initializing Collision Subsystem"));
237  collisions = new Collisions(m_sim_controller);
238  gEnv->collisions = collisions;
239 
240  PROGRESS_WINDOW(19, _L("Initializing Script Subsystem"));
241  initScripting();
242 
243  PROGRESS_WINDOW(21, _L("Initializing Shadow Subsystem"));
244 
245  PROGRESS_WINDOW(25, _L("Initializing Camera Subsystem"));
246  initCamera();

262  initVegetation();

264  // water must be done later on

269  {
270  PROGRESS_WINDOW(35, _L("Initializing HDR Subsystem"));

445 void TerrainManager::initMotionBlur()
446 {

461  def->formatList.push_back(Ogre::PF_R8G8B8);

478  tp->setInputMode(Ogre::CompositionTargetPass::IM_PREVIOUS);
479  tp->setOutputName("sum");

486  tp->setOutputName("temp");

488  Ogre::CompositionPass* pass = tp->createPass();
489  pass->setType(Ogre::CompositionPass::PT_RENDERQUAD);
490  pass->setMaterialName("Ogre/Compositor/Combine");

556 void TerrainManager::initWater()
557 {
558  // disabled in global config
559  if (App::gfx_water_mode.GetActive() == GfxWaterMode::NONE)
560  return;
561 
562  // disabled in map config
563  if (!m_def.has_water)
564  {
565  return;
566  }
567 
568  if (App::gfx_water_mode.GetActive() == GfxWaterMode::HYDRAX)
569  {
570  // try to load hydrax config
571  if (!m_def.hydrax_conf_file.empty() && ResourceGroupManager::getSingleton().resourceExistsInAnyGroup(m_def.hydrax_conf_file))
572  {
573  hw = new HydraxWater(m_def.water_height, m_def.hydrax_conf_file);
574  }
575  else
576  {
577  // no config provided, fall back to the default one
578  hw = new HydraxWater(m_def.water_height);
579  }
580 
581  water = hw;
582 
583  //Apply depth technique to the terrain
584  TerrainGroup::TerrainIterator ti = geometry_manager->getTerrainGroup()->getTerrainIterator();
585  while (ti.hasMoreElements())
586  {
587  Terrain* t = ti.getNext()->instance;
588  MaterialPtr ptr = t->getMaterial();
589  hw->GetHydrax()->getMaterialManager()->addDepthTechnique(ptr->createTechnique());
590  }
591  }
592  else
593  {
594  if (water == nullptr)
595  water = new Water();
596  else if (water != nullptr)
597  {
598  delete(water);
599  water = new Water();
600  }
601 
602  water->setHeight (m_def.water_height);
603  }
604 }
