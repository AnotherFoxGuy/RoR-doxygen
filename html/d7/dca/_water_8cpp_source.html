106  mReflectCam(0),
107  mRefractCam(0)
108 {
109  //Ugh.. Why so ugly and hard to read
110  mapSize = gEnv->terrainManager->getMaxTerrainSize();
111  waterSceneMgr = gEnv->sceneManager;
112 
113  if (mapSize.x < 1500 && mapSize.z < 1500)
114  mScale = 1.5f;
115 
116  char line[1024] = {};
117  std::string filepath = std::string(RoR::App::sys_config_dir.GetActive()) + PATH_SLASH + "wavefield.cfg";
118  FILE* fd = fopen(filepath.c_str(), "r");
119  if (fd)
120  {
121  while (!feof(fd))
122  {
123  int res = fscanf(fd, " %[^\n\r]", line);
124  if (line[0] == ';')
125  continue;
126  float wl, amp, mx, dir;
127  res = sscanf(line, "%f, %f, %f, %f", &wl, &amp, &mx, &dir);
128  if (res < 4)
129  continue;
130 
131  WaveTrain wavetrain;
132  wavetrain.wavelength = wl;
133  wavetrain.amplitude = amp;
134  wavetrain.maxheight = mx;
135  wavetrain.direction = dir / 57.0;
136  wavetrain.dir_sin = sin(wavetrain.direction);
137  wavetrain.dir_cos = cos(wavetrain.direction);
138 
139  m_wavetrain_defs.push_back(wavetrain);
140  }
141  fclose(fd);
142  }
143  for (size_t i = 0; i < m_wavetrain_defs.size(); i++)
144  {
145  m_wavetrain_defs[i].wavespeed = 1.25 * sqrt(m_wavetrain_defs[i].wavelength);
146  maxampl += m_wavetrain_defs[i].maxheight;
147  }
148 
149  this->processWater();
150 }

420 void Water::showWave(Vector3 refpos)
421 {
422  if (!wbuffer)
423  return;
424 
425  float xScaled = mapSize.x * mScale;
426  float zScaled = mapSize.z * mScale;
427 
428  for (int pz = 0; pz < WAVEREZ + 1; pz++)
429  {
430  for (int px = 0; px < WAVEREZ + 1; px++)
431  {
432  wbuffer[(pz * (WAVEREZ + 1) + px) * 8 + 1] = getHeightWaves(refpos + Vector3(xScaled * 0.5 - (float)px * xScaled / WAVEREZ, 0, (float)pz * zScaled / WAVEREZ - zScaled * 0.5)) - wHeight;
433  }
434  }
435 
436  //normals
437  for (int pz = 0; pz < WAVEREZ + 1; pz++)
438  {
439  for (int px = 0; px < WAVEREZ + 1; px++)
440  {
441  int left = std::max(0, px - 1);
442  int right = std::min(px + 1, WAVEREZ);
443  int up = std::max(0, pz - 1);
444  int down = std::min(pz + 1, WAVEREZ);
445 
446  Vector3 normal = (Vector3(wbuffer + ((pz * (WAVEREZ + 1) + left) * 8)) - Vector3(wbuffer + ((pz * (WAVEREZ + 1) + right) * 8))).crossProduct(Vector3(wbuffer + ((up * (WAVEREZ + 1) + px) * 8)) - Vector3(wbuffer + ((down * (WAVEREZ + 1) + px) * 8)));
447  normal.normalise();
448 
449  wbuffer[(pz * (WAVEREZ + 1) + px) * 8 + 3] = normal.x;
450  wbuffer[(pz * (WAVEREZ + 1) + px) * 8 + 4] = normal.y;
451  wbuffer[(pz * (WAVEREZ + 1) + px) * 8 + 5] = normal.z;
452  }
453  }
454 
455  wbuf->writeData(0, (WAVEREZ + 1) * (WAVEREZ + 1) * 32, wbuffer, true);
456 }

467 void Water::update()
468 {
469  if (!visible || !mRenderCamera)
470  return;
471 
472  if (pWaterNode)
473  {
474  Vector3 cameraPos(mRenderCamera->getPosition().x, wHeight, mRenderCamera->getPosition().z);
475  Vector3 sightPos(cameraPos);
476 
477  Ray lineOfSight(mRenderCamera->getPosition(), mRenderCamera->getDirection());
478  Plane waterPlane(Vector3::UNIT_Y, Vector3::UNIT_Y * wHeight);
479 
480  std::pair<bool, Real> intersection = lineOfSight.intersects(waterPlane);
481 
482  if (intersection.first && intersection.second > 0.0f)
483  sightPos = lineOfSight.getPoint(intersection.second);
484 
485  Real offset = std::min(cameraPos.distance(sightPos), std::min(mapSize.x, mapSize.z) * 0.5f);
486 
487  Vector3 waterPos = cameraPos + (sightPos - cameraPos).normalisedCopy() * offset;
488  Vector3 bottomPos = Vector3(waterPos.x, wbHeight, waterPos.z);
489 
490  if (waterPos.distance(pWaterNode->getPosition()) > 200.0f || ForceUpdate)
491  {
492  pWaterNode->setPosition(Vector3(waterPos.x, wHeight, waterPos.z));
493  pBottomNode->setPosition(bottomPos);
494  ForceUpdate = false; //Happens only once
495  }
496  if (RoR::App::gfx_water_waves.GetActive() && RoR::App::mp_state.GetActive() == RoR::MpState::DISABLED)
497  showWave(pWaterNode->getPosition());
498  }
499 
500  bool underwater = isCameraUnderWater();
501  static bool lastWaterMode = false;
502  bool underWaterModeChanged = false;
503  if (underwater != lastWaterMode)
504  {
505  underWaterModeChanged = true;
506  lastWaterMode = underwater;
507  }
508 
509  framecounter++;
510  if (App::gfx_water_mode.GetActive() == GfxWaterMode::FULL_FAST)
511  {
512  if (framecounter % 2)
513  {
514  mReflectCam->setOrientation(mRenderCamera->getOrientation());
515  mReflectCam->setPosition(mRenderCamera->getPosition());
516  mReflectCam->setFOVy(mRenderCamera->getFOVy());
517  rttTex2->update();
518  }
519  else
520  {
521  mRefractCam->setOrientation(mRenderCamera->getOrientation());
522  mRefractCam->setPosition(mRenderCamera->getPosition());
523  mRefractCam->setFOVy(mRenderCamera->getFOVy());
524  rttTex1->update();
525  }
526  }
527  else if (App::gfx_water_mode.GetActive() == GfxWaterMode::FULL_HQ)
528  {
529  mReflectCam->setOrientation(mRenderCamera->getOrientation());
530  mReflectCam->setPosition(mRenderCamera->getPosition());
531  mReflectCam->setFOVy(mRenderCamera->getFOVy());
532  rttTex2->update();
533  mRefractCam->setOrientation(mRenderCamera->getOrientation());
534  mRefractCam->setPosition(mRenderCamera->getPosition());
535  mRefractCam->setFOVy(mRenderCamera->getFOVy());
536  rttTex1->update();
537  }
538  else if (App::gfx_water_mode.GetActive() == GfxWaterMode::REFLECT)
539  {
540  mReflectCam->setOrientation(mRenderCamera->getOrientation());
541  mReflectCam->setPosition(mRenderCamera->getPosition());

566 float Water::getHeightWaves(Vector3 pos)
567 {
568  // no waves?
569  if (!RoR::App::gfx_water_waves.GetActive() || RoR::App::mp_state.GetActive() == RoR::MpState::CONNECTED)
570  {
571  // constant height, sea is flat as pancake
572  return wHeight;
573  }
574 
575  // uh, some upper limit?!
576  if (pos.y > wHeight + maxampl)
577  return wHeight;
578 
579  float waveheight = getWaveHeight(pos);
580  // we will store the result in this variable, init it with the default height
581  float result = wHeight;
582  // now walk through all the wave trains. One 'train' is one sin/cos set that will generate once wave. All the trains together will sum up, so that they generate a 'rough' sea
583  for (size_t i = 0; i < m_wavetrain_defs.size(); i++)
584  {
585  // calculate the amplitude that this wave will have. wavetrains[i].amplitude is read from the config
586  // upper limit: prevent too big waves by setting an upper limit
587  float amp = std::min(m_wavetrain_defs[i].amplitude * waveheight, m_wavetrain_defs[i].maxheight);
588  // now the main thing:
589  // calculate the sinus with the values of the config file and add it to the result
590  result += amp * sin(Math::TWO_PI * ((gEnv->mrTime * m_wavetrain_defs[i].wavespeed + m_wavetrain_defs[i].dir_sin * pos.x + m_wavetrain_defs[i].dir_cos * pos.z) / m_wavetrain_defs[i].wavelength));
591  }
592  // return the summed up waves
593  return result;
594 }

596 bool Water::isUnderWater(Vector3 pos)
597 {
598  float waterheight = wHeight;
599 
600  if (RoR::App::gfx_water_waves.GetActive() && RoR::App::mp_state.GetActive() == RoR::MpState::DISABLED)
601  {
602  float waveheight = getWaveHeight(pos);
603 
604  if (pos.y > wHeight + maxampl * waveheight || pos.y > wHeight + maxampl)
605  return false;

608  }

613 Vector3 Water::getVelocity(Vector3 pos)
614 {
615  if (!RoR::App::gfx_water_waves.GetActive() || RoR::App::mp_state.GetActive() == RoR::MpState::CONNECTED)
616  return Vector3::ZERO;
617
