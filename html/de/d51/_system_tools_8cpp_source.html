45 wxString LoadInputDevicesInfo(WXWidget wx_window_handle)
46 {
47  // Code cloned from original utility "inputtool.exe" located in ~REPO/tools/windows
48  // Useful reference:
49  // http://docs.wxwidgets.org/3.1/classwx_text_ctrl.html
50  // https://wiki.wxwidgets.org/Writing_Your_First_Application-Using_The_WxTextCtrl
51 
52  using namespace OIS;
53  const char *DeviceTypes[6] = {"OISUnknown", "OISKeyboard", "OISMouse", "OISJoyStick", "OISTablet", "OISOther"};
54  InputManager* input_man = nullptr;
55  wxString output;
56 
57  try
58  {
59  ParamList pl;
60  std::ostringstream wnd;
61  wnd << (size_t)wx_window_handle;
62  pl.insert(std::make_pair( std::string("WINDOW"), wnd.str() ));
63  input_man = InputManager::createInputSystem(pl);
64  input_man->enableAddOnFactory(InputManager::AddOn_All);
65  unsigned int v = input_man->getVersionNumber();
66  std::stringstream out_stream;
67  out_stream << "System info:\n\tOIS Version: " << (v>>16 ) << "." << ((v>>8) & 0x000000FF) << "." << (v & 0x000000FF)
68  << "\n\tOIS Release Name: " << input_man->getVersionName()
69  << "\n\tInput Manager: " << input_man->inputSystemName()
70  << "\n\tTotal Keyboards: " << input_man->getNumberOfDevices(OISKeyboard)
71  << "\n\tTotal Mice: " << input_man->getNumberOfDevices(OISMouse)
72  << "\n\tTotal JoySticks: " << input_man->getNumberOfDevices(OISJoyStick);
73  out_stream << "\n\nDevices:" << std::endl;
74  DeviceList list = input_man->listFreeDevices();
75 
76  for( DeviceList::iterator i = list.begin(); i != list.end(); ++i )
77  {
78  out_stream << "\t- " << DeviceTypes[i->first] << ", Vendor: " << i->second << std::endl;
79  }
80 
81  JoyStick* g_joys[50];
82  int numSticks = input_man->getNumberOfDevices(OISJoyStick);
83  for( int i = 0; i < numSticks; ++i )
84  {
85  g_joys[i] = (JoyStick*)input_man->createInputObject( OISJoyStick, true );
86  out_stream << "\n\nJoystick " << (i) << ":"
87  << "\n\tVendor: " << g_joys[i]->vendor()
88  << "\n\tVendorMapFilename: " << ComposeVendorMapFilename(g_joys[i]->vendor())
89  << "\n\tID: " << g_joys[i]->getID()
90  << "\n\tType: [" << g_joys[i]->type() << "] " << DeviceTypes[g_joys[i]->type()]
91  << "\n\tAxes: " << g_joys[i]->getNumberOfComponents(OIS_Axis)
92  << "\n\tSliders: " << g_joys[i]->getNumberOfComponents(OIS_Slider)
93  << "\n\tPOV/HATs: " << g_joys[i]->getNumberOfComponents(OIS_POV)
94  << "\n\tButtons: " << g_joys[i]->getNumberOfComponents(OIS_Button)
95  << "\n\tVector3: " << g_joys[i]->getNumberOfComponents(OIS_Vector3)
96  << "\n\tVector3Sensitivity: " << g_joys[i]->getVector3Sensitivity();
97  }
98 
99  output = wxString::FromUTF8(out_stream.str().c_str());
100  }

104  }

107  output = wxT("\n\tAn error occurred!");
108  }
109 
110  if(input_man != nullptr)
111  InputManager::destroyInputSystem(input_man);
112 
113  return output;
114 }

116 bool ExtractZipFiles(const wxString& aZipFile, const wxString& aTargetDir)
117 {
118  wxFileInputStream in(aZipFile);
119  if (! in.IsOk())
120  {
121  wxLogError(_T("Cannot open input ZIP file '") + aZipFile + wxT("'."));
122  return false;
123  }
124 
125  wxZipInputStream zip(in);
126  if (! zip.IsOk())
127  {
128  wxLogError(_T("Cannot read from ZIP file '") + aZipFile + wxT("'."));
129  return false;
130  }
131 
132  std::unique_ptr<wxZipEntry> entry; // Defaults to nullptr
133  while (entry.reset(zip.GetNextEntry()), entry.get() != nullptr)
134  {
135  if (entry->IsDir()) // We detect directories from file paths
136  {
137  continue;
138  }
139 
140  wxString entryName = entry->GetName();
141 
142  // Handle subdirectories
143  int lastSepPos = entryName.rfind(wxFileName::GetPathSeparator());
144  if (lastSepPos != -1)
145  {
146  wxString subdirPath = aTargetDir + wxFileName::GetPathSeparator() + entryName.SubString(0, lastSepPos);
147  const int permissions = 0777; // Equivalent of 'wxPosixPermissions::wxS_DIR_DEFAULT' in WxWidgets 3.x, we want to remain 2.x compatible.
148  // With flag 'wxPATH_MKDIR_FULL', existing directories are OK
149  if (! wxFileName::Mkdir(subdirPath, permissions, wxPATH_MKDIR_FULL))
150  {
151  wxLogError(_T("Could not create full subdirectory path '") + subdirPath + wxT("'."));
152  continue;
153  }
154  }
155 
156  // Skip dummy 'keep empty directory' files
157  if ((lastSepPos != -1) && (entryName.substr(lastSepPos+1) == wxString("empty")) && (entry->GetSize() == 0))
158  {
159  continue;
160  }
161 
162  // Extract the file
163  wxString targetName = aTargetDir + wxFileName::GetPathSeparator() + entryName;
164  wxFileOutputStream outFile(targetName);
165  if (! outFile.IsOk())
166  {
167  wxLogError(_T("Could not create output file '") + targetName + wxT("'."));
168  continue;
169  }
170  zip.Read(outFile); // Entry already open by 'GetNextEntry()'
171  }
172  return true;
173 }

177 std::string ComposeVendorMapFilename(std::string vendor)
178 {
179  std::string repl = "\\/ #@?!$%^&*()+=-><.:'|\";";
180  std::string vendorstr = std::string(vendor);
181  for(unsigned int c1 = 0; c1 < repl.size(); c1++)
182  for(unsigned int c2 = 0; c2 < vendorstr.size(); c2++)
183  if(vendorstr[c2] == repl[c1]) vendorstr[c2] = '_';
184  vendorstr += ".map";
185  return vendorstr;
186 }
