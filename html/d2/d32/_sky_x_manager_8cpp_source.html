79 bool SkyXManager::UpdateSkyLight()
80 {
81  Ogre::Vector3 lightDir = -getMainLightDirection();
82  Ogre::Vector3 sunPos = gEnv->mainCamera->getDerivedPosition() - lightDir*mSkyX->getMeshManager()->getSkydomeRadius(gEnv->mainCamera);
83 
84  // Calculate current color gradients point
85  float point = (-lightDir.y + 1.0f) / 2.0f;
86 
87  if (gEnv->terrainManager->getHydraxManager())
88  gEnv->terrainManager->getHydraxManager()->GetHydrax()->setWaterColor(mWaterGradient.getColor(point));
89 
90  mLight0 = gEnv->sceneManager->getLight("Light0");
91  mLight1 = gEnv->sceneManager->getLight("Light1");
92 
93  mLight0->setPosition(sunPos*0.02);
94  mLight1->setDirection(lightDir);
95  if (gEnv->terrainManager->getWater()) gEnv->terrainManager->getWater()->setSunPosition(sunPos*0.1);
96 
97  //setFadeColour was removed with https://github.com/RigsOfRods/rigs-of-rods/pull/1459
98 /* Ogre::Vector3 sunCol = mSunGradient.getColor(point);
99  mLight0->setSpecularColour(sunCol.x, sunCol.y, sunCol.z);
100  if (gEnv->terrainManager->getWater()) gEnv->terrainManager->getWater()->setFadeColour(Ogre::ColourValue(sunCol.x, sunCol.y, sunCol.z));
101  */
102  Ogre::Vector3 ambientCol = mAmbientGradient.getColor(point);
103  mLight1->setDiffuseColour(ambientCol.x, ambientCol.y, ambientCol.z);
104  mLight1->setPosition(100,100,100);
105 
106  if (mBasicController->getTime().x > 12)
107  {
108  if (mBasicController->getTime().x > mBasicController->getTime().z)
109  mLight0->setVisible(false);
110  else
111  mLight0->setVisible(true);
112  } else
113  if (mBasicController->getTime().x < mBasicController->getTime().z)
114  mLight0->setVisible(false);
115  else
116  mLight0->setVisible(true);
117 
118 
119  return true;
120 }
