76 void LanguageEngine::setup()
77 {
78 #ifdef USE_MOFILEREADER
79  // load language, must happen after initializing Settings class and Ogre Root!
80  // also it must happen after loading all basic resources!
81  reader = new moFileLib::moFileReader();
82 
83  Str<300> mo_path;
84  mo_path << App::sys_process_dir.GetActive() << PATH_SLASH << "languages" << PATH_SLASH;
85  mo_path << App::app_locale.GetActive()[0] << App::app_locale.GetActive()[1]; // Only first 2 chars are important
86  mo_path << PATH_SLASH << "LC_MESSAGES";
87 
88  // Load a .mo-File.
89  RoR::Log("[RoR|App] Loading language file...");
90  Str<300> rormo_path;
91  rormo_path << mo_path << PATH_SLASH << "ror.mo";
92  if (reader->ReadFile(rormo_path) != moFileLib::moFileReader::EC_SUCCESS)
93  {
94  RoR::LogFormat("[RoR|App] Error loading language file: '%s'", rormo_path.ToCStr());
95  return;
96  }
97  working = true;
98 
99  // add resource path
100  ResourceGroupManager::getSingleton().addResourceLocation(mo_path.GetBuffer(), "FileSystem", "LanguageFolder");
101 
102  ResourceGroupManager::getSingleton().initialiseResourceGroup("LanguageFolder");
103 
104  // now load the code ranges
105  // be aware, that this approach only works if we load just one language, and not multiple
106  setupCodeRanges("code_range.txt", "LanguageFolder");
107 #else
108  // init gettext
109  bindtextdomain("ror","languages");
110  textdomain("ror");
111 
112  char *curr_locale = setlocale(LC_ALL, NULL);
113  if (curr_locale)
114  RoR::LogFormat("[RoR|App] System locale is '%s'", curr_locale);
115  else
116  RoR::Log("[RoR|App] Warning: Unable to detect system locale!");
117 
118  String language_short = Ogre::String(App::app_locale.GetActive()).substr(0, 2); // only first two characters are important
119  if (!language_short.empty())
120  {
121  RoR::LogFormat("[RoR|App] Setting locale to '%s'", language_short.c_str());
122  char *newlocale = setlocale(LC_ALL, language_short.c_str());
123  if (newlocale)
124  RoR::LogFormat("[RoR|App] New locale is '%s'", newlocale);
125  else
126  RoR::Log("[RoR|App] Error setting system locale");
127  }
128  else
129  {
130  RoR::Log("[RoR|App] Using system locale without change");
131  }
132 
133 #endif // USE_MOFILEREADER
134  RoR::Log("[RoR|App] Language successfully loaded");
135 }
