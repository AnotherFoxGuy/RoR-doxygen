153 void Settings::loadSettings(String configFile, bool overwrite)
154 {
155  ConfigFile cfg;
156  cfg.load(configFile, "=:\t", false);
157 
158  // load all settings into a map!
159  ConfigFile::SettingsIterator i = cfg.getSettingsIterator();
160  String s_value, s_name;
161  while (i.hasMoreElements())
162  {
163  s_name = SanitizeUtf8String(i.peekNextKey());
164  s_value = SanitizeUtf8String(i.getNext());
165  if (!overwrite && !settings[s_name].empty())
166  {
167  continue;
168  }
169  settings[s_name] = s_value;
170  }
171 }

173 bool Settings::get_system_paths(char *program_path, char *user_path)
174 {
175 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32
176  // note: we enforce usage of the non-UNICODE interfaces (since its easier to integrate here)
177  if (!GetModuleFileNameA(NULL, program_path, 512))
178  {
179  ErrorUtils::ShowError(_L("Startup error"), _L("Error while retrieving program space path"));
180  return false;
181  }
182  GetShortPathNameA(program_path, program_path, 512); //this is legal
183  path_descend(program_path);
184 
185  if (getSetting("userpath", "").empty())
186  {
187  if (SHGetFolderPathA(NULL, CSIDL_PERSONAL, NULL, SHGFP_TYPE_CURRENT, user_path)!=S_OK)
188  {
189  ErrorUtils::ShowError(_L("Startup error"), _L("Error while retrieving user space path"));
190  return false;
191  }
192  GetShortPathNameA(user_path, user_path, 512); //this is legal
193  sprintf(user_path, "%s\\Rigs of Rods %s\\", user_path, ROR_VERSION_STRING_SHORT); // do not use the full string, as same minor versions should share this directory
194  } else
195  {
196  strcpy(user_path, getSetting("userpath", "").c_str());
197  }
198 
199 #elif OGRE_PLATFORM == OGRE_PLATFORM_LINUX
200  //true program path is impossible to get from POSIX functions
201  //lets hack!
202  pid_t pid = getpid();
203  char procpath[256];
204  sprintf(procpath, "/proc/%d/exe", pid);
205  int ch = readlink(procpath,program_path,240);
206  if (ch != -1)
207  {
208  program_path[ch] = 0;
209  path_descend(program_path);
210  } else return false;
211  //user path is easy
212  char home_path[256];
213  strncpy(home_path, getenv ("HOME"), 240);
214  //sprintf(user_path, "%s/RigsOfRods/", home_path); // old version
215  sprintf(user_path, "%s/.rigsofrods/", home_path);
216 #elif OGRE_PLATFORM == OGRE_PLATFORM_APPLE
217  int cx;
218 
219  // program path
220  const int len = 256;
221  uint32_t lenb = (uint32_t) len;
222  char procpath[len];
223 
224  if (_NSGetExecutablePath(procpath, &lenb) == -1)
225  return false;
226 
227  /*
228  * _NSGetExecutablePath returns the absolute path to the binary so procpath
229  * has "./RoR" at its end. We only want the path to the directory so we
230  * cut off the last 5 characters of the string.
231  */
232  assert(strlen(procpath) > 5);
233  procpath[strlen(procpath) - 4] = '\0';
234 
235  cx = snprintf(program_path, len, "%s", procpath);
236  if ( cx < 0 || cx >= len)
237  return false;
238 
239  // user path
240  cx = snprintf(user_path, 256, "%s/RigsOfRods/", getenv("HOME"));
241  if ( cx < 0 || cx >= 256)
242  return false;
243 #endif
244  return true;
245 }

247 bool Settings::setupPaths()
248 {
249  char program_path[1024] = {};
250  //char streams_path[1024] = {};
251  char user_path[1024] = {};
252 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32
253  const char *dsStr = "\\";
254 #else
255  const char *dsStr="/";
256 #endif
257 
258  if (!get_system_paths(program_path, user_path))
259  return false;
260 
261  String local_config = String(program_path) + String(dsStr) + String("config");
262  if (FolderExists(local_config.c_str()))
263  {
264  sprintf(user_path, "%s%sconfig%s",program_path, dsStr, dsStr);
265  }
266 
267  // change working directory to executable path
268 #ifdef _WIN32
269  _chdir(program_path);
270 #endif // _WIN32
271 
272 
273  // now update our settings with the results:
274 
275  // only set log path if it was not set before
276  m_user_path = user_path;
277  m_program_path = program_path;
278 
279  return true;
280 }

283 void Settings::path_descend(char* path)
284 {
285  char dirsep='/';
286 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32
287  dirsep='\\';
288 #endif
289  char* pt=path+strlen(path)-1;
290  if (pt>=path && *pt==dirsep) pt--;
291  while (pt>=path && *pt!=dirsep) pt--;
292  if (pt>=path) *(pt+1)=0;
293 }
