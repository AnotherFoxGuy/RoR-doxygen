128  void VCloudsManager::_setLightParameters()
129  {
130  Ogre::Vector3 SunDir = -mSkyX->getController()->getSunDirection();
131 
132  // Moon
133  if (SunDir.y > 0.175f)
134  {

137 
138  mVClouds->setSunDirection(SunDir);
139 
140  float point = (mSkyX->getController()->getSunDirection().y + 1.0f) / 2.0f;
141 
142  mVClouds->setAmbientColor(mAmbientGradient.getColor(point));
143  mVClouds->setSunColor(mSunGradient.getColor(point));
