161 void GetParentDirectory(char* dst_buf, const char* src_buff)
162 {
163  const char* start = src_buff;
164  size_t count = strlen(src_buff);
165  // Trim trailing separator(s)
166  for (;;)
167  {
168  if (count == 0) { dst_buf[0] = '\0'; return; }
169  if (!IsSeparator(start[count - 1])) { break; }
170  --count;
171  }
172  // Remove last path entry
173  for (;;)
174  {
175  if (count == 0) { dst_buf[0] = '\0'; return; }
176  if (IsSeparator(start[count - 1])) {break; }
177  --count;
178  }
179  // Trim rear separator(s)
180  for (;;)
181  {
182  if (count == 0) { dst_buf[0] = '\0'; return; }
183  if (!IsSeparator(start[count - 1])) { break; }
184  --count;
185  }
186  strncpy(dst_buf, start, count);
187 }

189 int DetectBasePaths()
190 {
191  constexpr int bufsize = 500;
192  char buf[bufsize] = "";
193 
194  // Process dir (system)
195 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32 // NOTE: We use non-UNICODE interfaces for simplicity
196  // Process dir
197  if (!GetModuleFileNameA(nullptr, buf, bufsize))
198  {
199  return -1;
200  }
201 
202 #elif OGRE_PLATFORM == OGRE_PLATFORM_LINUX // http://stackoverflow.com/a/625523
203  memset(buf, 0, bufsize);
204  // Process dir
205  if (readlink("/proc/self/exe", buf, bufsize-1) == -1)
206  {
207  return -1;
208  }
209 
210 #elif OGRE_PLATFORM == OGRE_PLATFORM_APPLE
211  // Process dir
212  uint32_t length = bufsize;
213  if (_NSGetExecutablePath(procpath, &length) == -1) // Returns absolute path to binary
214  {
215  return -1;
216  }
217 #endif
218  Str<bufsize> process_dir;
219  GetParentDirectory(process_dir.GetBuffer(), buf);
220  App::sys_process_dir.SetActive(process_dir);
221 
222  // User directory (local override - portable installation)
223  Str<bufsize> local_userdir;
224  local_userdir << App::sys_process_dir.GetActive() << PATH_SLASH << "config";
225  if (FolderExists(local_userdir))
226  {
227  App::sys_user_dir.SetActive(local_userdir);
228  return 0; // Done!
229  }
230 
231  // User directory (system)
232 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32 // NOTE: We use non-UNICODE interfaces for simplicity
233  if (SHGetFolderPathA(nullptr, CSIDL_PERSONAL, nullptr, SHGFP_TYPE_CURRENT, buf) != S_OK)
234  {
235  return -2;
236  }
237  sprintf(buf, "%s\\Rigs of Rods %s", buf, ROR_VERSION_STRING_SHORT);
238 
239 #elif OGRE_PLATFORM == OGRE_PLATFORM_LINUX
240  snprintf(buf, bufsize, "%s/.rigsofrods", getenv("HOME"));
241 
242 #elif OGRE_PLATFORM == OGRE_PLATFORM_APPLE
243  snprintf(buf, bufsize, "%s/RigsOfRods", getenv("HOME"));
244 
245 #endif
246 
247  App::sys_user_dir.SetActive(buf);
248  return 0;
249 }

256 void Settings::ProcessCommandLine(int argc, char *argv[])
257 {
258  CSimpleOpt args(argc, argv, cmdline_options);
259 
260  while (args.Next())
261  {
262  if (args.LastError() != SO_SUCCESS)
263  {
264  RoR::App::app_state.SetPending(RoR::AppState::PRINT_HELP_EXIT);
265  return;
266  }
267  else if (args.OptionId() == OPT_HELP)
268  {
269  RoR::App::app_state.SetPending(RoR::AppState::PRINT_HELP_EXIT);
270  return;
271  }
272  else if (args.OptionId() == OPT_VER)
273  {
274  RoR::App::app_state.SetPending(RoR::AppState::PRINT_VERSION_EXIT);
275  return;
276  }
277  else if (args.OptionId() == OPT_TRUCK)
278  {
279  App::diag_preset_vehicle.SetActive(args.OptionArg());
280  }
281  else if (args.OptionId() == OPT_TRUCKCONFIG)
282  {
283  App::diag_preset_veh_config.SetActive(args.OptionArg());
284  }
285  else if (args.OptionId() == OPT_MAP)
286  {
287  App::diag_preset_terrain.SetActive(args.OptionArg());
288  }
289  else if (args.OptionId() == OPT_USERPATH)
290  {
291  SETTINGS.setSetting("userpath", String(args.OptionArg()));
292  }
293  else if (args.OptionId() == OPT_WDIR)
294  {
295 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32
296  SetCurrentDirectory(args.OptionArg());
297 #endif
298  }
299  else if (args.OptionId() == OPT_STATE)
300  {
301  SETTINGS.setSetting("StartState", args.OptionArg());
302  }
303  else if (args.OptionId() == OPT_NOCACHE)
304  {
305  SETTINGS.setSetting("NOCACHE", "Yes");
306  }
307  else if (args.OptionId() == OPT_INCLUDEPATH)
308  {
309  App::diag_extra_resource_dir.SetActive(args.OptionArg());
310  }
311  else if (args.OptionId() == OPT_CHECKCACHE)
312  {
313  // just regen cache and exit
314  SETTINGS.setSetting("regen-cache-only", "Yes");
315  }
316  else if (args.OptionId() == OPT_ENTERTRUCK)
317  {
318  App::diag_preset_veh_enter.SetActive(true);
319  }
320  else if (args.OptionId() == OPT_SETUP)
321  {
322  SETTINGS.setSetting("USE_OGRE_CONFIG", "Yes");
323  }
324  else if (args.OptionId() == OPT_JOINMPSERVER)
325  {
326  std::string server_args = args.OptionArg();
327  const int colon = server_args.rfind(":");
328  if (colon != std::string::npos)
329  {
330  App::mp_state.SetPending(RoR::MpState::CONNECTED);
331 
332  std::string host_str;
333  std::string port_str;
334  if (server_args.find("rorserver://") != String::npos) // Windows URI Scheme retuns rorserver://server:port/
335  {
336  host_str = server_args.substr(12, colon - 12);
337  port_str = server_args.substr(colon + 1, server_args.length() - colon - 2);
338  }
339  else
340  {
341  host_str = server_args.substr(0, colon);
342  port_str = server_args.substr(colon + 1, server_args.length());
343  }
344  App::mp_server_host.SetActive(host_str.c_str());
345  App::mp_server_port.SetActive(Ogre::StringConverter::parseInt(port_str));
346  }
347  }
348  }

822  {
823  Str<300> path;
824  path << App::sys_config_dir.GetActive() << PATH_SLASH << "RoR.cfg";
825  cfg.load(Ogre::String(path), "=:\t", false);
826 
827  // load all settings into a map!
828  Ogre::ConfigFile::SettingsIterator i = cfg.getSettingsIterator();
829  String s_value, s_name;
830  while (i.hasMoreElements())
831  {
832  s_name = RoR::Utils::SanitizeUtf8String(i.peekNextKey());
833  s_value = RoR::Utils::SanitizeUtf8String(i.getNext());
834 
835  // Purge unwanted entries
836  if (s_name == "Program Path") { continue; }
837 
838  // Process and clear GVar values
839  if (this->ParseGlobalVarSetting(s_name, s_value))
840  {
841  continue;
842  }
843 
844  settings[s_name] = s_value;
845  }
846  }
847  catch (Ogre::FileNotFoundException&) {} // Just continue with defaults...
848 }
