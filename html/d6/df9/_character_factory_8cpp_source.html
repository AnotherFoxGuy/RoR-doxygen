81 {
82  for (auto packet : packet_buffer)
83  {
84  if (packet.header.command == RoRnet::MSG2_STREAM_REGISTER)

86  RoRnet::StreamRegister* reg = (RoRnet::StreamRegister *)packet.buffer;
87  if (reg->type == 1)
88  {
89  createRemoteInstance(packet.header.source, packet.header.streamid);
90  }
91  }
92  else if (packet.header.command == RoRnet::MSG2_USER_LEAVE)
93  {
94  removeStreamSource(packet.header.source);
95  }
96  else
97  {
98  for (auto& c : m_remote_characters)
99  {
100  c->receiveStreamData(packet.header.command, packet.header.source, packet.header.streamid, packet.buffer);
101  }
102  }
103  }
