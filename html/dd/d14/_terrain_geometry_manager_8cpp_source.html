149 void TerrainGeometryManager::getTerrainPositionAlign(Real x, Real y, Real z, Terrain::Alignment align, Vector3* outTSpos)
150 {
151  switch (align)
152  {
153  case Terrain::ALIGN_X_Z:
154  outTSpos->x = (x - mBase - mPos.x) / ((mSize - 1) * mScale);
155  outTSpos->y = (z + mBase - mPos.z) / ((mSize - 1) * -mScale);
156  outTSpos->z = y;
157  break;
158  case Terrain::ALIGN_Y_Z:
159  outTSpos->x = (z - mBase - mPos.z) / ((mSize - 1) * -mScale);
160  outTSpos->y = (y + mBase - mPos.y) / ((mSize - 1) * mScale);
161  outTSpos->z = x;
162  break;
163  case Terrain::ALIGN_X_Y:
164  outTSpos->x = (x - mBase - mPos.x) / ((mSize - 1) * mScale);
165  outTSpos->y = (y - mBase - mPos.y) / ((mSize - 1) * mScale);
166  outTSpos->z = z;
167  break;
168  };
169 }

184 float TerrainGeometryManager::getHeightAtTerrainPosition(Real x, Real y)
185 {
186  // get left / bottom points (rounded down)
187  Real factor = (Real)mSize - 1.0f;
188  Real invFactor = 1.0f / factor;
189 
190  long startX = static_cast<long>(x * factor);
191  long startY = static_cast<long>(y * factor);
192  long endX = startX + 1;
193  long endY = startY + 1;
194 
195  // now get points in terrain space (effectively rounding them to boundaries)
196  // note that we do not clamp! We need a valid plane
197  Real startXTS = startX * invFactor;
198  Real startYTS = startY * invFactor;
199  Real endXTS = endX * invFactor;
200  Real endYTS = endY * invFactor;
201 
202  // now clamp
203  endX = std::min(endX, (long)mSize - 1);
204  endY = std::min(endY, (long)mSize - 1);
205 
206  // get parametric from start coord to next point
207  Real xParam = (x - startXTS) / invFactor;
208  Real yParam = (y - startYTS) / invFactor;
209 
210  /* For even / odd tri strip rows, triangles are this shape:
211  even odd
212  3---2 3---2
213  | / | | \ |
214  0---1 0---1
215  */
216 
217  // Build all 4 positions in terrain space, using point-sampled height
218  Vector3 v0(startXTS, startYTS, getHeightAtPoint(startX, startY));
219  Vector3 v1(endXTS, startYTS, getHeightAtPoint(endX, startY));
220  Vector3 v2(endXTS, endYTS, getHeightAtPoint(endX, endY));
221  Vector3 v3(startXTS, endYTS, getHeightAtPoint(startX, endY));
222  // define this plane in terrain space
223  Plane plane;
224  if (startY % 2)
225  {
226  // odd row
227  bool secondTri = ((1.0 - yParam) > xParam);
228  if (secondTri)
229  plane.redefine(v0, v1, v3);
230  else
231  plane.redefine(v1, v2, v3);
232  }
233  else
234  {
235  // even row
236  bool secondTri = (yParam > xParam);
237  if (secondTri)
238  plane.redefine(v0, v2, v3);
239  else
240  plane.redefine(v0, v1, v2);
241  }
242 
243  // Solve plane equation for z
244  return (-plane.normal.x * x
245  - plane.normal.y * y
246  - plane.d) / plane.normal.z;
247 }

270 Ogre::Vector3 TerrainGeometryManager::getNormalAt(float x, float y, float z, float precision)
271 {
272  Ogre::Vector3 left(-precision, getHeightAt(x - precision, z) - y, 0.0f);
273  Ogre::Vector3 down(0.0f, getHeightAt(x, z + precision) - y, precision);
274  down = left.crossProduct(down);
275  down.normalise();
276  return down;

279 void TerrainGeometryManager::InitTerrain(std::string otc_filename)
280 {
281  try
282  {
283  OTCParser otc_parser;
284  DataStreamPtr ds_config = ResourceGroupManager::getSingleton().openResource(otc_filename);
285  otc_parser.LoadMasterConfig(ds_config, otc_filename.c_str());
286 
287  for (OTCPage& page : otc_parser.GetDefinition()->pages)
288  {
289  if (page.pageconf_filename.empty())
290  continue;
291 
292  DataStreamPtr ds_page = ResourceGroupManager::getSingleton().openResource(page.pageconf_filename);
293  if (ds_page.isNull() || !ds_page->isReadable())
294  {
295  LOG("[RoR|Terrain] Cannot read file [" + page.pageconf_filename + "].");
296  continue;
297  }
298 
299  otc_parser.LoadPageConfig(ds_page, page, page.pageconf_filename.c_str());
300  }
301 
302  m_spec = otc_parser.GetDefinition();
303  }
304  catch(...) // Error already reported
305  {
306  return;
307  }
308 
309  const std::string cache_filename_format = m_spec->cache_filename_base + "_OGRE_" + TOSTRING(OGRE_VERSION) + "_";
310 
311  m_ogre_terrain_group = OGRE_NEW TerrainGroup(gEnv->sceneManager, Terrain::ALIGN_X_Z, m_spec->page_size, m_spec->world_size);
312  m_ogre_terrain_group->setFilenameConvention(cache_filename_format, "mapbin");
313  m_ogre_terrain_group->setOrigin(m_spec->origin_pos);
314  m_ogre_terrain_group->setResourceGroup("cache");
315 
316  configureTerrainDefaults();
317 
318  auto* loading_win = RoR::App::GetGuiManager()->GetLoadingWindow();
319  for (OTCPage& page : m_spec->pages)
320  {
321  loading_win->setProgress(23, _L("preparing terrain page ") + XZSTR(page.pos_x, page.pos_z));
322  this->SetupGeometry(page, m_spec->is_flat);
323  }
324 
325  // sync load since we want everything in place when we start
326  loading_win->setProgress(23, _L("loading terrain pages"));
327  m_ogre_terrain_group->loadAllTerrains(true);
328 
329  Terrain* terrain = m_ogre_terrain_group->getTerrain(0, 0);
330  mHeightData = terrain->getHeightData();
331  mAlign = terrain->getAlignment();
332  mSize = terrain->getSize();
333  const float world_size = terrain->getWorldSize();
334  mBase = -world_size * 0.5f;
335  mScale = world_size / (Real)(mSize - 1);
336  mPos = terrain->getPosition();
337 
338  // update the blend maps
339  if (m_was_new_geometry_generated)
340  {
341  if (m_terrain_mgr->GetDef().custom_material_name.empty())
342  {
343  for (OTCPage& page : m_spec->pages)
344  {
345  Ogre::Terrain* terrain = m_ogre_terrain_group->getTerrain(page.pos_x, page.pos_z);
346 
347  if (terrain != nullptr)
348  {
349  loading_win->setProgress(23, _L("loading terrain page layers ") + XZSTR(page.pos_x, page.pos_z));
350  this->SetupLayers(page, terrain);
351  loading_win->setProgress(23, _L("loading terrain page blend maps ") + XZSTR(page.pos_x, page.pos_z));
352  this->SetupBlendMaps(page, terrain);
353  }
354  }
355  }
356 
357  // always save the results when it was imported
358  if (!m_spec->disable_cache)
359  {
360  loading_win->setProgress(23, _L("saving all terrain pages ..."));
361  m_ogre_terrain_group->saveAllTerrains(false);
362  }
363  }
364  else
365  {
366  LOG(" *** Terrain loaded from cache ***");
367  }
368 
369  m_ogre_terrain_group->freeTemporaryResources();
370 }

527 void TerrainGeometryManager::SetupBlendMaps(OTCPage& page, Ogre::Terrain* terrain )
528 {
529  const int layerCount = terrain->getLayerCount();
530  auto layer_def_itor = page.layers.begin();
531  ++layer_def_itor;
532  for (int i = 1; i < layerCount; i++)
533  {
534  if (layer_def_itor->blendmap_filename.empty())
535  continue;
536 
537  Ogre::Image img;
538  try
539  {
540  img.load(layer_def_itor->blendmap_filename, ResourceGroupManager::AUTODETECT_RESOURCE_GROUP_NAME);
541  }
542  catch (Exception& e)
543  {
544  LOG("Error loading blendmap: " + layer_def_itor->blendmap_filename + " : " + e.getFullDescription());
545  continue;
546  }
547 
548  TerrainLayerBlendMap* blendmap = terrain->getLayerBlendMap(i);
549 
550  // resize that blending map so it will fit
551  const Ogre::uint32 blendmapSize = terrain->getLayerBlendMapSize();
552  if (img.getWidth() != blendmapSize)
553  img.resize(blendmapSize, blendmapSize);
554 
555  // now to the ugly part
556  float* ptr = blendmap->getBlendPointer();
557  for (Ogre::uint32 z = 0; z != blendmapSize; z++)
558  {
559  for (Ogre::uint32 x = 0; x != blendmapSize; x++)
560  {
561  Ogre::ColourValue c = img.getColourAt(x, z, 0);
562  const float alpha = layer_def_itor->alpha;
563  if (layer_def_itor->blend_mode == 'R')
564  *ptr++ = c.r * alpha;
565  else if (layer_def_itor->blend_mode == 'G')
566  *ptr++ = c.g * alpha;
567  else if (layer_def_itor->blend_mode == 'B')
568  *ptr++ = c.b * alpha;
569  else if (layer_def_itor->blend_mode == 'A')
570  *ptr++ = c.a * alpha;
571  }
572  }
573  blendmap->dirty();
574  blendmap->update();
575  ++layer_def_itor;
576  }
577 
578  if (m_spec->blendmap_dbg_enabled)
579  {
580  for (int i = 1; i < layerCount; i++)
581  {
582  Ogre::TerrainLayerBlendMap* blendMap = terrain->getLayerBlendMap(i);
583  Ogre::uint32 blendmapSize = terrain->getLayerBlendMapSize();
584  Ogre::Image img;
585  unsigned short* idata = OGRE_ALLOC_T(unsigned short, blendmapSize * blendmapSize, Ogre::MEMCATEGORY_RESOURCE);
586  float scale = 65535.0f;
587  for (unsigned int x = 0; x < blendmapSize; x++)
588  for (unsigned int z = 0; z < blendmapSize; z++)
589  idata[x + z * blendmapSize] = (unsigned short)(blendMap->getBlendValue(x, blendmapSize - z) * scale);
590  img.loadDynamicImage((Ogre::uchar*)(idata), blendmapSize, blendmapSize, Ogre::PF_L16);
591  std::string fileName = "blendmap_layer_" + Ogre::StringConverter::toString(i) + ".png";
592  img.save(fileName);
593  OGRE_FREE(idata, Ogre::MEMCATEGORY_RESOURCE);
594  }
595  }
596 }

600 {
601  if (page.heightmap_filename.empty())
602  {
603  LOG("[RoR|Terrain] Empty Heightmap provided in OTC, please use 'Flat=1' instead");

605  }
606

611  LOG("[RoR|Terrain] loading RAW image: " + TOSTRING(stream->size()) + " / " + TOSTRING(page.raw_size*page.raw_size*page.raw_bpp));

613  img.loadRawData(stream, page.raw_size, page.raw_size, 1, pix_format);
614  }
615  else
616  {
617  img.load(page.heightmap_filename, ResourceGroupManager::DEFAULT_RESOURCE_GROUP_NAME);
618  }
619 
620  if (page.raw_flip_x)
621  img.flipAroundX();
622  if (page.raw_flip_y)
623  img.flipAroundY();
624 
625  return true;
