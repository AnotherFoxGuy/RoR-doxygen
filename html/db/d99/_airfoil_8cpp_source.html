29 Airfoil::Airfoil(Ogre::String const& fname)
30 {
31  for (int i = 0; i < 3601; i++) //init in case of bad things
32  {
33  cl[i] = 0;
34  cd[i] = 0;
35  cm[i] = 0;
36  }
37  char line[1024];
38  //we load directly X-Plane AFL file format!!!
39  bool process = false;
40  bool neg = true;
41  int lastia = -1;
42  ResourceGroupManager& rgm = ResourceGroupManager::getSingleton();
43 
44  String group = "";
45  try
46  {
47  group = rgm.findGroupContainingResource(fname);
48  }
49  catch (...)
50  {
51  }
52  if (group == "")
53  {
54  LOG(String("Airfoil error: could not load airfoil ")+fname);
55  return;
56  }
57 
58  DataStreamPtr ds = rgm.openResource(fname, group);
59  while (!ds->eof())
60  {
61  size_t ll = ds->readLine(line, 1023);
62  if (ll == 0)
63  continue;
64  // fscanf(fd," %[^\n\r]",line);
65  if (!strncmp("alpha", line, 5))
66  {
67  process = true;
68  continue;
69  };
70  if (process)
71  {
72  float l, d, m;
73  int a, b;
74  sscanf(line, "%i.%i %f %f %f", &a, &b, &l, &d, &m);
75  if (neg)
76  b = -b;
77  if (a == 0 && b == 0)
78  neg = false;
79  int ia = (a * 10 + b) + 1800;
80  if (ia == 3600) { process = false; };
81  cl[ia] = l;
82  cd[ia] = d;
83  cm[ia] = m;
84  if (lastia != -1 && ia - lastia > 1)
85  {
86  //we have to interpolate previous elements (linear interpolation)
87  int i;
88  for (i = 0; i < ia - lastia - 1; i++)
89  {
90  cl[lastia + 1 + i] = cl[lastia] + (float)(i + 1) * (cl[ia] - cl[lastia]) / (float)(ia - lastia);
91  cd[lastia + 1 + i] = cd[lastia] + (float)(i + 1) * (cd[ia] - cd[lastia]) / (float)(ia - lastia);
92  cm[lastia + 1 + i] = cm[lastia] + (float)(i + 1) * (cm[ia] - cm[lastia]) / (float)(ia - lastia);
93  }
94  }
95  lastia = ia;
96  }
97  }
98 }

104 void Airfoil::getparams(float a, float cratio, float cdef, float* ocl, float* ocd, float* ocm)
105 {
106  int ta = (int)(a / 360.0);
107  // float va=360.0f*fmod(a, 360.0f); FMOD IS TOTALLY UNRELIABLE HERE : fmod(-180.0f, 360.0f)=-180.0f!!!!!
108  float va = a - (float)(ta * 360);
109  if (va > 180.0f)
110  va -= 360.0f;
111  if (va < -180.0f)
112  va += 360.0f;
113  int ia = (int)((va + 180.0f) * 10.0f);
114  //drag shift
115  float dva = va + 1.15 * (1.0 - cratio) * cdef;
116  if (dva > 180.0f)
117  dva -= 360.0f;
118  if (dva < -180.0f)
119  dva += 360.0f;
120  int dia = (int)((dva + 180.0f) * 10.0f);
121  float sign = 1.0;
122  if (cdef < 0)
123  sign = -1.0;
124  *ocl = cl[ia] - 0.66 * sign * (1.0 - cratio) * sqrt(fabs(cdef));
125  *ocd = cd[dia] + 0.00015 * (1.0 - cratio) * cdef * cdef;
126  *ocm = cm[ia] + 0.20 * sign * (1.0 - cratio) * sqrt(fabs(cdef));
127 }
