153 GUIManager::GUIManager() :
154  m_renderwindow_closed(false),
155  m_impl(nullptr)
156 {
157  RoR::App::GetOgreSubsystem()->GetOgreRoot()->addFrameListener(this);
158  RoRWindowEventUtilities::addWindowEventListener(RoR::App::GetOgreSubsystem()->GetRenderWindow(), this);
159 
160  Str<300> gui_logpath;
161  gui_logpath << App::sys_logs_dir.GetActive() << PATH_SLASH << "MyGUI.log";
162  auto mygui_platform = new MyGUI::OgrePlatform();
163  mygui_platform->initialise(
164  RoR::App::GetOgreSubsystem()->GetRenderWindow(),
165  gEnv->sceneManager,
166  Ogre::ResourceGroupManager::AUTODETECT_RESOURCE_GROUP_NAME,
167  gui_logpath.GetBuffer()); // use cache resource group so preview images are working
168  auto mygui = new MyGUI::Gui();
169 
170  // empty init
171  mygui->initialise("");
172 
173  // add layer factory
174  MyGUI::FactoryManager::getInstance().registerFactory<MyGUI::RTTLayer>("Layer");
175 
176  // then load the actual config
177  MyGUI::ResourceManager::getInstance().load(RESOURCE_FILENAME);
178 
179  MyGUI::ResourceManager::getInstance().load(LanguageEngine::getSingleton().getMyGUIFontConfigFilename());
180 
181  m_impl = new GuiManagerImpl();
182  m_impl->mygui_platform = mygui_platform;
183  m_impl->mygui = mygui;
184  MyGUI::PointerManager::getInstance().setVisible(false); // RoR is using mouse cursor drawn by DearIMGUI.
185 
186 #ifdef _WIN32
187  MyGUI::LanguageManager::getInstance().eventRequestTag = MyGUI::newDelegate(this, &GUIManager::eventRequestTag);
188 #endif // _WIN32
189  windowResized(RoR::App::GetOgreSubsystem()->GetRenderWindow());
190 
191  this->SetupImGui();
192 }

286 Ogre::String GUIManager::getRandomWallpaperImage()
287 {
288  using namespace Ogre;
289  FileInfoListPtr files = ResourceGroupManager::getSingleton().findResourceFileInfo("Wallpapers", "*.jpg", false);
290  if (files.isNull() || files->empty())
291  {
292  files = ResourceGroupManager::getSingleton().findResourceFileInfo("Wallpapers", "*.png", false);
293  if (files.isNull() || files->empty())
294  return "";
295  }
296  srand ( time(NULL) );
297 
298  int num = 0;
299  for (int i = 0; i<Math::RangeRandom(0, 10); i++)
300  num = Math::RangeRandom(0, files->size());
301 
302  return files->at(num).filename;
303 }

408 void GUIManager::ReflectGameState()
409 {
410  const auto app_state = App::app_state.GetActive();
411  const auto mp_state = App::mp_state.GetActive();
412  if (app_state == AppState::MAIN_MENU)
413  {
414  m_impl->panel_GameMainMenu .SetVisible(!m_impl->panel_MainSelector.IsVisible());
415 
416  m_impl->panel_TopMenubar .SetVisible(false);
417  m_impl->panel_ChatBox .SetVisible(false);
418  m_impl->panel_DebugOptions .SetVisible(false);
419  m_impl->panel_FrictionSettings .SetVisible(false);
420  m_impl->panel_GamePauseMenu .SetVisible(false);
421  m_impl->panel_TextureToolWindow .SetVisible(false);
422  m_impl->panel_TeleportWindow .SetVisible(false);
423  m_impl->panel_VehicleDescription .SetVisible(false);
424  m_impl->panel_SpawnerReport .SetVisible(false);
425  m_impl->panel_SimUtils .SetBaseVisible(false);
426  m_impl->panel_MpClientList .SetVisible(mp_state == MpState::CONNECTED);
427  return;
428  }
429  if (app_state == AppState::SIMULATION)
430  {
431  m_impl->panel_TopMenubar .SetVisible(true);
432  m_impl->panel_TopMenubar .ReflectMultiplayerState();

434  m_impl->panel_GameMainMenu .SetVisible(false);
435  return;
436  }
