86 float Autopilot::getAilerons()
87 {
88  float val = 0;
89  if (ref_l && ref_r)
90  {
91  float rat = (ref_r->RelPosition.y - ref_l->RelPosition.y) / ref_span;
92  float bank = 90.0;
93  if (rat >= 1.0)
94  bank = 90.0;
95  else if (rat <= -1.0)
96  bank = -90.0;
97  else
98  bank = 57.3 * asin(rat);
99  if (mode_heading == HEADING_WLV)
100  {
101  val = bank / 100.0;
102  if (val > 0.5)
103  val = 0.5;
104  if (val < -0.5)
105  val = -0.5;
106  }
107  if (mode_heading == HEADING_FIXED)
108  {
109  Vector3 vel = (ref_l->Velocity + ref_r->Velocity) / 2.0;
110  float curdir = atan2(vel.x, -vel.z) * 57.295779513082;
111  float want_bank = curdir - (float)heading;
112  if (want_bank < -180.0)
113  want_bank += 360.0;
114  want_bank = want_bank * 2.0;
115  if (want_bank > 45.0)
116  want_bank = 45.0;
117  if (want_bank < -45.0)
118  want_bank = -45.0;
119  val = (bank - want_bank) / 100.0;
120  if (val > 0.5)
121  val = 0.5;
122  if (val < -0.5)
123  val = -0.5;
124  }
125  if (mode_heading == HEADING_NAV)
126  {
127  //compute intercept heading
128  float error_heading = m_ils_angle_hdev / 10.0;
129  if (error_heading > 1.0)
130  error_heading = 1.0;
131  if (error_heading < -1.0)
132  error_heading = -1.0;
133  float offcourse_tolerance = m_ils_runway_distance / 30.0;
134  if (offcourse_tolerance > 60.0)
135  offcourse_tolerance = 60.0;
136  float intercept_heading = m_ils_runway_heading + error_heading * offcourse_tolerance;
137  Vector3 vel = (ref_l->Velocity + ref_r->Velocity) / 2.0;
138  float curdir = atan2(vel.x, -vel.z) * 57.295779513082;
139  float want_bank = curdir - intercept_heading;
140  if (want_bank < -180.0)
141  want_bank += 360.0;
142  want_bank = want_bank * 2.0;
143  if (want_bank > 45.0)
144  want_bank = 45.0;
145  if (want_bank < -45.0)
146  want_bank = -45.0;
147  val = (bank - want_bank) / 100.0;
148  if (val > 0.5)
149  val = 0.5;
150  if (val < -0.5)
151  val = -0.5;
152  }
153  }
154  last_aileron = (last_aileron + val) / 2.0;
155  return last_aileron;
156 }

158 float Autopilot::getElevator()
159 {
160  float val = 0;
161  if (ref_l && ref_r && ref_b)
162  {
163  float wanted_vs = (float)vs / 196.87;
164  float current_vs = (ref_l->Velocity.y + ref_r->Velocity.y) / 2.0;
165  float pitch_var = current_vs - ref_b->Velocity.y;
166  if (mode_alt == ALT_VS)
167  {
168  if (mode_heading == HEADING_NAV)
169  {
170  //this is cheating a bit
171  float ch = m_ils_runway_distance * sin((m_ils_angle_vdev + 4.0) / 57.295779513082);
172  float oh = m_ils_runway_distance * sin((4.0) / 57.295779513082);
173  wanted_vs = 5000.0 / 196.87;
174  float wanted_vs2 = (-ch + oh) / 5.0;
175  if (wanted_vs2 < -wanted_vs)
176  wanted_vs2 = -wanted_vs;
177  if (wanted_vs2 > wanted_vs)
178  wanted_vs2 = wanted_vs;
179  val = (wanted_vs2 - current_vs) / 40.0 + pitch_var / 40.0;
180  if (val > 0.75)
181  val = 0.75;
182  if (val < -0.75)
183  val = -0.75;
184  }
185  else
186  {
187  val = (wanted_vs - current_vs) / 40.0 + pitch_var / 40.0;
188  if (val > 0.5)
189  val = 0.5;
190  if (val < -0.5)
191  val = -0.5;
192  }
193  }
194  if (mode_alt == ALT_FIXED)
195  {
196  float wanted_alt = (float)alt * 0.3048;
197  float current_alt = (ref_l->AbsPosition.y + ref_r->AbsPosition.y) / 2.0;
198  if (wanted_vs < 0)
199  wanted_vs = -wanted_vs; //absolute value
200  float wanted_vs2 = (wanted_alt - current_alt) / 8.0;
201  if (wanted_vs2 < -wanted_vs)
202  wanted_vs2 = -wanted_vs;
203  if (wanted_vs2 > wanted_vs)
204  wanted_vs2 = wanted_vs;
205  val = (wanted_vs2 - current_vs) / 40.0 + pitch_var / 40.0;
206  if (val > 0.5)
207  val = 0.5;
208  if (val < -0.5)
209  val = -0.5;
210  }
211  }
212  return val;
213 }

220 float Autopilot::getThrottle(float thrtl, float dt)
221 {
222  if (!mode_ias) { return thrtl; };
223 
224  float val = thrtl;
225  if (ref_l && ref_r)
226  {
227  //tropospheric model valid up to 11.000m (33.000ft)
228  float altitude = ref_l->AbsPosition.y;
229  //float sea_level_temperature=273.15+15.0; //in Kelvin
230  float sea_level_pressure = 101325; //in Pa
231  //float airtemperature=sea_level_temperature-altitude*0.0065; //in Kelvin
232  float airpressure = sea_level_pressure * pow(1.0 - 0.0065 * altitude / 288.15, 5.24947); //in Pa
233  float airdensity = airpressure * 0.0000120896;//1.225 at sea level
234 
235  float gspd = 1.94384449 * ((ref_l->Velocity + ref_r->Velocity) / 2.0).length();
236 
237  float spd = gspd * sqrt(airdensity / 1.225); //KIAS
238 
239  if (spd > (float)ias)
240  val = val - dt / 15.0;
241  if (spd < (float)ias)
242  val = val + dt / 15.0;
243  if (val > 1.0)
244  val = 1.0;
245  if (val < 0.02)
246  val = 0.02;
247  }
248  return val;
249 }

317 void Autopilot::gpws_update(float spawnheight)
318 {
319 #ifdef USE_OPENAL
320  if (SoundScriptManager::getSingleton().isDisabled())
321  return;
322  if (mode_gpws && gEnv->terrainManager->getHeightFinder() && ref_b)
323  {
324  float groundalt = gEnv->terrainManager->getHeightFinder()->getHeightAt(ref_c->AbsPosition.x, ref_c->AbsPosition.z);
325  if (gEnv->terrainManager->getWater() && groundalt < gEnv->terrainManager->getWater()->GetStaticWaterHeight())
326  groundalt = gEnv->terrainManager->getWater()->GetStaticWaterHeight();
327  float height = (ref_c->AbsPosition.y - groundalt - spawnheight) * 3.28083f; //in feet!
328  //skip height warning sounds when the plane is slower then ~10 knots
329  if ((ref_c->Velocity.length() * 1.9685f) > 10.0f)
330  {
331  if (height < 10 && last_gpws_height > 10)
332  SOUND_PLAY_ONCE(m_actor_id, SS_TRIG_GPWS_10);
333  if (height < 20 && last_gpws_height > 20)
334  SOUND_PLAY_ONCE(m_actor_id, SS_TRIG_GPWS_20);
335  if (height < 30 && last_gpws_height > 30)
336  SOUND_PLAY_ONCE(m_actor_id, SS_TRIG_GPWS_30);
337  if (height < 40 && last_gpws_height > 40)
338  SOUND_PLAY_ONCE(m_actor_id, SS_TRIG_GPWS_40);
339  if (height < 50 && last_gpws_height > 50)
340  SOUND_PLAY_ONCE(m_actor_id, SS_TRIG_GPWS_50);
341  if (height < 100 && last_gpws_height > 100)
342  SOUND_PLAY_ONCE(m_actor_id, SS_TRIG_GPWS_100);
343  }
344  last_gpws_height = height;
345 
346  // pullup warning calculation
347  // height to meters
348  height *= 0.3048;
349  // get the y-velocity in meters/s
350  float yVel = ref_c->Velocity.y * 1.9685f;
351  // will trigger the pullup sound when vvi is high (avoid pullup warning when landing normal) and groundcontact will be in less then 10 seconds
352  if (yVel * 10.0f < -height && yVel < -10.0f)
353  SOUND_PLAY_ONCE(m_actor_id, SS_TRIG_GPWS_PULLUP);
354  }
355 #endif //OPENAL
356 }

358 void Autopilot::UpdateIls(std::vector<TerrainObjectManager::localizer_t> localizers)
359 {
360  if (!ref_l || !ref_r)
361  return;
362  Vector3 position = (ref_l->AbsPosition + ref_r->AbsPosition) / 2.0;
363  float closest_hdist = -1;
364  float closest_hangle = -90;
365  float closest_vdist = -1;
366  float closest_vangle = -90;
367  m_ils_runway_heading = 0;
368  for (std::vector<TerrainObjectManager::localizer_t>::size_type i = 0; i < localizers.size(); i++)
369  {
370  Plane hplane = Plane(Vector3::UNIT_Y, 0);
371  Vector3 plocd = hplane.projectVector(localizers[i].rotation * Vector3::UNIT_Z);
372  float loc = atan2(plocd.z, plocd.x);
373  Vector3 posd = hplane.projectVector(position - localizers[i].position);
374  float dir = atan2(posd.z, posd.x);
375  float diff = (dir - loc) * 57.295779513082;
376  if (diff > 180.0)
377  diff -= 360.0;
378  if (diff < -180.0)
379  diff += 360.0;
380  if (diff < 80 && diff > -80)
381  {
382  //we are in the visibility cone, we search the closest
383  float dist = (localizers[i].position - position).length();
384  //horizontal
385  if (localizers[i].type == Autopilot::LOCALIZER_HORIZONTAL)
386  {
387  if (closest_hdist < 0 || closest_hdist > dist)
388  {
389  closest_hdist = dist;
390  closest_hangle = diff;
391  m_ils_runway_heading = loc * 57.295779513082 - 90.0; //dont ask me why
392  if (m_ils_runway_heading < 0.0)
393  m_ils_runway_heading += 360.0;
394  m_ils_runway_distance = dist;
395  }
396  }
397  //vertical
398  if (localizers[i].type == Autopilot::LOCALIZER_VERTICAL)
399  {
400  if (closest_vdist < 0 || closest_vdist > dist)
401  {
402  Vector3 normv = (localizers[i].rotation * Vector3::UNIT_Z).crossProduct(Vector3::UNIT_Y);
403  Plane vplane = Plane(normv, 0);
404  float glideslope = 4; //4 degree is the norm
405  Vector3 posd2 = vplane.projectVector(position - localizers[i].position);
406  float d = posd2.length();
407  if (d < 0.01)
408  d = 0.01;
409  float dir2 = 90.0;
410  d = posd2.y / d;
411  if (d >= 1.0)
412  dir2 = 90.0;
413  else if (d <= -1.0)
414  dir2 = -90.0;
415  else
416  dir2 = asin(d) * 57.295779513082;
417  float diff2 = dir2 - glideslope;
418  closest_vdist = dist;
419  closest_vangle = diff2;
420  }
421  }
422  }
423  }
424  m_ils_angle_hdev = closest_hangle;
425  m_ils_angle_vdev = closest_vangle;
426 
427  m_horizontal_locator_available = (closest_hdist != -1);
428  m_vertical_locator_available = (closest_vdist != -1);
429 
430  if (mode_heading == HEADING_NAV && mode_gpws && closest_hdist > 10.0 && closest_hdist < 350.0 && last_closest_hdist > 10.0 && last_closest_hdist > 350.0)
431  {
432  SOUND_PLAY_ONCE(m_actor_id, SS_TRIG_GPWS_MINIMUMS);
433  }
434 
435  last_closest_hdist = closest_hdist;
436  if (mode_heading == HEADING_NAV)
437  {
438  // disconnect if close to runway or no locators are available
439  if (closest_hdist < 20.0 || closest_vdist < 20.0)
440  wantsdisconnect = true;
441  if (!this->IsIlsAvailable())
442  wantsdisconnect = true;
443  }
