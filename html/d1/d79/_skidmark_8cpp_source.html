37 void RoR::SkidmarkConfig::LoadDefaultSkidmarkDefs()
38 {
39  LOG("[RoR] Loading skidmarks.cfg...");
40  Ogre::String group = "";
41  try
42  {
43  group = Ogre::ResourceGroupManager::getSingleton().findGroupContainingResource("skidmarks.cfg");
44  if (group.empty())
45  {
46  LOG("[RoR] Failed to load skidmarks.cfg (file not found)");
47  return;
48  }
49 
50  Ogre::DataStreamPtr ds = Ogre::ResourceGroupManager::getSingleton().openResource("skidmarks.cfg", group);
51  Ogre::String line = "";
52  Ogre::String currentModel = "";
53 
54  while (!ds->eof())
55  {
56  line = RoR::Utils::SanitizeUtf8String(ds->getLine());
57  Ogre::StringUtil::trim(line);
58 
59  if (line.empty() || line[0] == ';')
60  continue;
61 
62  Ogre::StringVector args = Ogre::StringUtil::split(line, ",");
63 
64  if (args.size() == 1)
65  {
66  currentModel = line;
67  continue;
68  }
69 
70  // process the line if we got a model
71  if (!currentModel.empty())
72  this->ProcessSkidmarkConfLine(args, currentModel);
73  }
74  }
75  catch (...)
76  {
77  LOG("[RoR] Error loading skidmarks.cfg (unknown error)");
78  m_models.clear(); // Delete anything we might have loaded
79  return;
80  }
81  LOG("[RoR] skidmarks.cfg loaded OK");
82 }

84 int RoR::SkidmarkConfig::ProcessSkidmarkConfLine(Ogre::StringVector args, Ogre::String modelName)
85 {
86  // we only accept 4 arguments
87  if (args.size() != 4)
88  return 1;
89 
90  // parse the data
91  SkidmarkDef cfg;
92  cfg.ground = args[0];
93  Ogre::StringUtil::trim(cfg.ground);
94  cfg.texture = args[1];
95  Ogre::StringUtil::trim(cfg.texture);
96 
97  cfg.slipFrom = Ogre::StringConverter::parseReal(args[2]);
98  cfg.slipTo = Ogre::StringConverter::parseReal(args[3]);
99 
100  if (!m_models.size() || m_models.find(modelName) == m_models.end())
101  m_models[modelName] = std::vector<SkidmarkDef>();
102 
103  m_models[modelName].push_back(cfg);
104  return 0;
105 }

159 void RoR::Skidmark::AddObject(Ogre::Vector3 start, Ogre::String texture)
160 {
161  SkidmarkSegment skid;
162  skid.pos = 0;
163  skid.lastPointAv = start;
164  skid.facecounter = 0;
165 
166  // new material
167  char bname[256] = "";
168  sprintf(bname, "mat-skidmark-%d", m_instance_counter);
169  skid.material = Ogre::MaterialManager::getSingleton().create(bname, Ogre::ResourceGroupManager::DEFAULT_RESOURCE_GROUP_NAME);
170  Ogre::Pass* p = skid.material->getTechnique(0)->getPass(0);
171 
172  Ogre::TextureUnitState* tus = p->createTextureUnitState(texture);
173  p->setSceneBlending(Ogre::SBT_TRANSPARENT_ALPHA);
174  p->setLightingEnabled(false);
175  p->setDepthWriteEnabled(false);
176  p->setDepthBias(3, 3);
177  p->setCullingMode(Ogre::CULL_NONE);
178 
179  skid.points.resize(m_length);
180  skid.faceSizes.resize(m_length);
181  skid.groundTexture.resize(m_length);
182  skid.obj = gEnv->sceneManager->createManualObject("skidmark" + TOSTRING(m_instance_counter++));
183  skid.obj->setDynamic(true);
184  skid.obj->setRenderingDistance(800); // 800m view distance
185  skid.obj->begin(bname, Ogre::RenderOperation::OT_TRIANGLE_STRIP);
186  for (int i = 0; i < m_length; i++)
187  {
188  skid.points[i] = start;
189  skid.faceSizes[i] = 0;
190  skid.groundTexture[i] = "0.png";
191  skid.obj->position(start);
192  skid.obj->textureCoord(0, 0);
193  }
194  skid.obj->end();
195  m_scene_node->attachObject(skid.obj);
196 
197  m_objects.push(skid);
198 
199  this->LimitObjects();
200 }

230 void RoR::Skidmark::updatePoint()
231 {
232  Ogre::Vector3 thisPoint = m_wheel->lastContactType ? m_wheel->lastContactOuter : m_wheel->lastContactInner;
233  Ogre::Vector3 axis = m_wheel->lastContactType ? (m_wheel->refnode1->RelPosition - m_wheel->refnode0->RelPosition) : (m_wheel->refnode0->RelPosition - m_wheel->refnode1->RelPosition);
234  Ogre::Vector3 thisPointAV = thisPoint + axis * 0.5f;
235  Ogre::Real distance = 0;
236  Ogre::Real maxDist = m_max_distance;
237  Ogre::String texture = "none";
238  m_config->getTexture("default", m_wheel->lastGroundModel->name, m_wheel->lastSlip, texture);
239 
240  // dont add points with no texture
241  if (texture == "none")
242  return;
243 
244  if (m_wheel->speed > 1)
245  maxDist *= m_wheel->speed;
246 
247  if (!m_objects.size())
248  {
249  // add first bucket
250  this->AddObject(thisPoint, texture);
251  }
252  else
253  {
254  // check existing buckets
255  SkidmarkSegment skid = m_objects.back();
256 
257  distance = skid.lastPointAv.distance(thisPointAV);
258  // too near to update?
259  if (distance < m_min_distance)
260  {
261  //LOG("E: too near for update");
262  return;
263  }
264 
265  // change ground texture if required
266  if (skid.pos > 0 && skid.groundTexture[0] != texture)
267  {
268  // new object with new texture
269  if (distance > maxDist)
270  {
271  // to far away for connection
272  this->AddObject(thisPoint, texture);
273  }
274  else
275  {
276  // add new bucket with connection to last bucket
277  Ogre::Vector3 lp1 = m_objects.back().points[m_objects.back().pos - 1];
278  Ogre::Vector3 lp2 = m_objects.back().points[m_objects.back().pos - 2];
279  this->AddObject(lp1, texture);
280  this->AddPoint(lp2, distance, texture);
281  this->AddPoint(lp1, distance, texture);
282  }
283  }
284  else
285  {
286  // no texture change required :D
287 
288  // far enough for new bucket?
289  if (skid.pos >= (int)skid.points.size())
290  {
291  if (distance > maxDist)
292  {
293  // to far away for connection
294  this->AddObject(thisPoint, texture);
295  }
296  else
297  {
298  // add new bucket with connection to last bucket
299  Ogre::Vector3 lp1 = m_objects.back().points[m_objects.back().pos - 1];
300  Ogre::Vector3 lp2 = m_objects.back().points[m_objects.back().pos - 2];
301  this->AddObject(lp1, texture);
302  this->AddPoint(lp2, distance, texture);
303  this->AddPoint(lp1, distance, texture);
304  }
305  }
306  else if (distance > m_max_distance)
307  {
308  // just new bucket, no connection to last bucket
309  this->AddObject(thisPoint, texture);
310  }
311  }
312  }
313 
314  const float overaxis = 0.2f;
315  // tactics: we always choose the latest point and then create two points
316 
317  // choose node m_wheel by the latest added point
318  if (!m_wheel->lastContactType)
319  {
320  // choose inner
321  this->AddPoint(m_wheel->lastContactInner - (axis * overaxis), distance, texture);
322  this->AddPoint(m_wheel->lastContactInner + axis + (axis * overaxis), distance, texture);
323  }
324  else
325  {
326  // choose outer
327  this->AddPoint(m_wheel->lastContactOuter + axis + (axis * overaxis), distance, texture);
328  this->AddPoint(m_wheel->lastContactOuter - (axis * overaxis), distance, texture);
329  }
330 
331  // save as last point (in the middle of the m_wheel)
332  m_objects.back().lastPointAv = thisPointAV;
333 }

345 void RoR::Skidmark::update()
346 {
347  if (!m_objects.size())
348  return;
349  SkidmarkSegment skid = m_objects.back();
350  Ogre::Vector3 vaabMin = skid.points[0];
351  Ogre::Vector3 vaabMax = skid.points[0];
352  skid.obj->beginUpdate(0);
353  bool behindEnd = false;
354  Ogre::Vector3 lastValid = Ogre::Vector3::ZERO;
355  int to_counter = 0;
356  float tcox_counter = 0;
357 
358  for (int i = 0; i < m_length; i++ , to_counter++)
359  {
360  if (i >= skid.pos)
361  behindEnd = true;
362 
363  if (to_counter > 3)
364  to_counter = 0;
365 
366  if (!behindEnd)
367  tcox_counter += skid.faceSizes[i] / m_min_distance;
368 
369  while (tcox_counter > 1)
370  tcox_counter--;
371 
372  if (behindEnd)
373  {
374  skid.obj->position(lastValid);
375  skid.obj->textureCoord(0, 0);
376  }
377  else
378  {
379  skid.obj->position(skid.points[i]);
380 
381  Ogre::Vector2 tco = m_tex_coords[to_counter];
382  tco.x *= skid.faceSizes[i] / m_min_distance; // scale texture according face size
383  skid.obj->textureCoord(tco);
384 
385  lastValid = skid.points[i];
386  }
387 
388  if (skid.points[i].x < vaabMin.x)
389  vaabMin.x = skid.points[i].x;
390  if (skid.points[i].y < vaabMin.y)
391  vaabMin.y = skid.points[i].y;
392  if (skid.points[i].z < vaabMin.z)
393  vaabMin.z = skid.points[i].z;
394  if (skid.points[i].x > vaabMax.x)
395  vaabMax.x = skid.points[i].x;
396  if (skid.points[i].y > vaabMax.y)
397  vaabMax.y = skid.points[i].y;
398  if (skid.points[i].z > vaabMax.z)
399  vaabMax.z = skid.points[i].z;
400  }
401  skid.obj->end();
402 
403  skid.obj->setBoundingBox(Ogre::AxisAlignedBox(vaabMin, vaabMax));
404 
405  m_is_dirty = false;
406 }
